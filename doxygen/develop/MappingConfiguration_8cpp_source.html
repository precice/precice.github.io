343 void MappingConfiguration::xmlTagCallback(
344  const xml::ConfigurationContext &context,
345  xml::XMLTag & tag)
346 {
347  PRECICE_TRACE(tag.getName());
348  if (tag.getNamespace() == TAG) {
349  // Mandatory tags
350  std::string dir = tag.getStringAttributeValue(ATTR_DIRECTION);
351  std::string fromMesh = tag.getStringAttributeValue(ATTR_FROM);
352  std::string toMesh = tag.getStringAttributeValue(ATTR_TO);
353  std::string type = tag.getName();
354  std::string constraint = tag.getStringAttributeValue(ATTR_CONSTRAINT);
355 
356  // optional tags
357  // We set here default values, but their actual value doesn't really matter.
358  // It's just for the mapping methods, which do not use these attributes at all.
359  bool xDead = tag.getBooleanAttributeValue(ATTR_X_DEAD, false);
360  bool yDead = tag.getBooleanAttributeValue(ATTR_Y_DEAD, false);
361  bool zDead = tag.getBooleanAttributeValue(ATTR_Z_DEAD, false);
362  double solverRtol = tag.getDoubleAttributeValue(ATTR_SOLVER_RTOL, 1e-9);
363  std::string strPolynomial = tag.getStringAttributeValue(ATTR_POLYNOMIAL, POLYNOMIAL_SEPARATE);
364 
365  // pum related tags
366  int verticesPerCluster = tag.getIntAttributeValue(ATTR_VERTICES_PER_CLUSTER, 100);
367  double relativeOverlap = tag.getDoubleAttributeValue(ATTR_RELATIVE_OVERLAP, 0.3);
368  bool projectToInput = tag.getBooleanAttributeValue(ATTR_PROJECT_TO_INPUT, true);
369 
370  // Convert raw string into enum types as the constructors take enums
371  if (constraint == CONSTRAINT_CONSERVATIVE) {
372  constraintValue = Mapping::CONSERVATIVE;
373  } else if (constraint == CONSTRAINT_CONSISTENT) {
374  constraintValue = Mapping::CONSISTENT;
375  } else if (constraint == CONSTRAINT_SCALED_CONSISTENT_SURFACE) {
376  constraintValue = Mapping::SCALED_CONSISTENT_SURFACE;
377  } else if (constraint == CONSTRAINT_SCALED_CONSISTENT_VOLUME) {
378  constraintValue = Mapping::SCALED_CONSISTENT_VOLUME;
379  } else {
380  PRECICE_UNREACHABLE("Unknown mapping constraint \"{}\".", constraint);
381  }
382 
383  ConfiguredMapping configuredMapping = createMapping(dir, type, fromMesh, toMesh);
384 
385  _rbfConfig = configureRBFMapping(type, strPolynomial, xDead, yDead, zDead, solverRtol, verticesPerCluster, relativeOverlap, projectToInput);
386 
387  checkDuplicates(configuredMapping);
388  _mappings.push_back(configuredMapping);
389  } else if (tag.getNamespace() == SUBTAG_BASIS_FUNCTION) {
390 
391  PRECICE_ASSERT(!_mappings.empty());
392  PRECICE_CHECK(_mappings.back().requiresBasisFunction == true, "A basis-function was defined for the mapping "
393  "from mesh \"{}\" to mesh \"{}\", but no basis-function is required for this mapping type. "
394  "Please remove the basis-function tag or configure an rbf mapping, which requires a basis-function.",
395  _mappings.back().fromMesh->getName(), _mappings.back().toMesh->getName());
396  // We can only set one subtag
397  PRECICE_CHECK(_rbfConfig.basisFunctionDefined == false, "More than one basis-function was defined for the mapping "
398  "from mesh \"{}\" to mesh \"{}\".",
399  _mappings.back().fromMesh->getName(), _mappings.back().toMesh->getName());
400 
401  std::string basisFctName = tag.getName();
402  double supportRadius = tag.getDoubleAttributeValue(ATTR_SUPPORT_RADIUS, 0.);
403  double shapeParameter = tag.getDoubleAttributeValue(ATTR_SHAPE_PARAM, 0.);
404 
405  _rbfConfig.basisFunction = parseBasisFunctions(basisFctName);
406  _rbfConfig.basisFunctionDefined = true;
407  // The Gaussian RBF is always treated as a shape-parameter RBF. Hence, we have to convert the support radius, if necessary
408  if (_rbfConfig.basisFunction == BasisFunction::Gaussian) {
409  const bool exactlyOneSet = (std::isfinite(supportRadius) && !std::isfinite(shapeParameter)) ||
410  (std::isfinite(shapeParameter) && !std::isfinite(supportRadius));
411  PRECICE_CHECK(exactlyOneSet, "The specified parameters for the Gaussian RBF mapping are invalid. Please specify either a \"shape-parameter\" or a \"support-radius\".");
412 
413  if (std::isfinite(supportRadius) && !std::isfinite(shapeParameter)) {
414  shapeParameter = std::sqrt(-std::log(Gaussian::cutoffThreshold)) / supportRadius;
415  }
416  }
417 
418  _rbfConfig.supportRadius = supportRadius;
419  _rbfConfig.shapeParameter = shapeParameter;
420  } else if (tag.getNamespace() == SUBTAG_EXECUTOR) {
421  _executorConfig = std::make_unique<ExecutorConfiguration>();
422 
423  if (tag.getName() == EXECUTOR_CPU) {
424  _executorConfig->executor = ExecutorConfiguration::Executor::CPU;
425  } else if (tag.getName() == EXECUTOR_CUDA) {
426  _executorConfig->executor = ExecutorConfiguration::Executor::CUDA;
427  } else if (tag.getName() == EXECUTOR_HIP) {
428  _executorConfig->executor = ExecutorConfiguration::Executor::HIP;
429  } else if (tag.getName() == EXECUTOR_OMP) {
430  _executorConfig->executor = ExecutorConfiguration::Executor::OpenMP;
431  }
432 
433  _executorConfig->deviceId = tag.getIntAttributeValue(ATTR_DEVICE_ID, -1);
434  _executorConfig->nThreads = tag.getIntAttributeValue(ATTR_N_THREADS, 0);
435  }

443  double relativeOverlap,
444  bool projectToInput) const
445 {
446  RBFConfiguration rbfConfig;
447 
448  if (type == TYPE_RBF_GLOBAL_ITERATIVE)
449  rbfConfig.solver = RBFConfiguration::SystemSolver::GlobalIterative;
450  else if (type == TYPE_RBF_GLOBAL_DIRECT)
451  rbfConfig.solver = RBFConfiguration::SystemSolver::GlobalDirect;
452  else if (type == TYPE_RBF_PUM_DIRECT)
453  rbfConfig.solver = RBFConfiguration::SystemSolver::PUMDirect;
454  else {
455  // Rather simple auto-selection (for now), consisting of the PUM Eigen backend
456  rbfConfig.solver = RBFConfiguration::SystemSolver::PUMDirect;
457 
458  // A more sophisticated criterion here could take the globalNumberOfVertices into account
459  // (the mesh pointer is stored in the configuredMapping anyway), but this quantity is not yet computed
460  // during the configuration time.
461  }
462 
463  if (polynomial == POLYNOMIAL_SEPARATE)
464  rbfConfig.polynomial = Polynomial::SEPARATE;
465  else if (polynomial == POLYNOMIAL_ON)
466  rbfConfig.polynomial = Polynomial::ON;
467  else if (polynomial == POLYNOMIAL_OFF)
468  rbfConfig.polynomial = Polynomial::OFF;
469  else
470  PRECICE_UNREACHABLE("Unknown polynomial configuration.");
471 
472  rbfConfig.deadAxis = {{xDead, yDead, zDead}};
473  rbfConfig.solverRtol = solverRtol;
474 
475  rbfConfig.verticesPerCluster = verticesPerCluster;
476  rbfConfig.relativeOverlap = relativeOverlap;
477  rbfConfig.projectToInput = projectToInput;
478 
479  return rbfConfig;

482 MappingConfiguration::ConfiguredMapping MappingConfiguration::createMapping(
483  const std::string &direction,
484  const std::string &type,
485  const std::string &fromMeshName,
486  const std::string &toMeshName) const
487 {
488  PRECICE_TRACE(direction, type);
489 
490  ConfiguredMapping configuredMapping;
491  mesh::PtrMesh fromMesh(_meshConfig->getMesh(fromMeshName));
492  mesh::PtrMesh toMesh(_meshConfig->getMesh(toMeshName));
493  PRECICE_CHECK(fromMesh.get() != nullptr,
494  "Mesh \"{0}\" was not found while creating a mapping. "
495  "Please correct the from=\"{0}\" attribute.",
496  fromMeshName);
497  PRECICE_CHECK(toMesh.get() != nullptr,
498  "Mesh \"{0}\" was not found while creating a mapping. "
499  "Please correct the to=\"{0}\" attribute.",
500  toMeshName);
501 
502  // Check for compatible mesh dimensions
503  PRECICE_CHECK(fromMesh->getDimensions() == toMesh->getDimensions(),
504  "Mapping between meshes of different dimensions is not allowed yet. "
505  "Please set the same dimensions attribute to meshes \"{}\" and \"{}\", "
506  "or choose different meshes.",
507  fromMeshName, toMeshName);
508 
509  configuredMapping.fromMesh = fromMesh;
510  configuredMapping.toMesh = toMesh;
511 
512  if (direction == DIRECTION_WRITE) {
513  configuredMapping.direction = WRITE;
514  } else if (direction == DIRECTION_READ) {
515  configuredMapping.direction = READ;
516  } else {
517  PRECICE_UNREACHABLE("Unknown mapping direction type \"{}\".", direction);
518  }
519 
520  // Create all projection based mappings
521  if (type == TYPE_NEAREST_NEIGHBOR) {
522  configuredMapping.mapping = PtrMapping(new NearestNeighborMapping(constraintValue, fromMesh->getDimensions()));
523  } else if (type == TYPE_NEAREST_PROJECTION) {
524  configuredMapping.mapping = PtrMapping(new NearestProjectionMapping(constraintValue, fromMesh->getDimensions()));
525  } else if (type == TYPE_LINEAR_CELL_INTERPOLATION) {
526  configuredMapping.mapping = PtrMapping(new LinearCellInterpolationMapping(constraintValue, fromMesh->getDimensions()));
527  } else if (type == TYPE_NEAREST_NEIGHBOR_GRADIENT) {
528 
529  // NNG is not applicable with the conservative constraint
530  PRECICE_CHECK(constraintValue != Mapping::CONSERVATIVE,
531  "Nearest-neighbor-gradient mapping is not implemented using a \"conservative\" constraint. "
532  "Please select constraint=\" consistent\" or a different mapping method.");
533 
534  configuredMapping.mapping = PtrMapping(new NearestNeighborGradientMapping(constraintValue, fromMesh->getDimensions()));
535  } else {
536  // We need knowledge about the basis function in order to instantiate the rbf related mapping
537  PRECICE_ASSERT(requiresBasisFunction(type));
538  configuredMapping.mapping = nullptr;
539  configuredMapping.configuredWithAliasTag = type == TYPE_RBF_ALIAS;
540  }
541 
542  configuredMapping.requiresBasisFunction = requiresBasisFunction(type);
543 
544  return configuredMapping;
545 }

575 void MappingConfiguration::finishRBFConfiguration()
576 {
577  PRECICE_ASSERT(_executorConfig);
578  ConfiguredMapping &mapping = _mappings.back();
579  // Instantiate the RBF mapping classes
580  // We first categorize according to the executor
581  // 1. the CPU executor
582  if (_executorConfig->executor == ExecutorConfiguration::Executor::CPU) {
583  if (_rbfConfig.solver == RBFConfiguration::SystemSolver::GlobalDirect) {
584  mapping.mapping = getRBFMapping<RBFBackend::Eigen>(_rbfConfig.basisFunction, constraintValue, mapping.fromMesh->getDimensions(), _rbfConfig.supportRadius, _rbfConfig.shapeParameter, _rbfConfig.deadAxis, _rbfConfig.polynomial);
585  } else if (_rbfConfig.solver == RBFConfiguration::SystemSolver::GlobalIterative) {
586 #ifndef PRECICE_NO_PETSC
587  // for petsc initialization
588  utils::Petsc::initialize(utils::Parallel::current()->comm);
589 
590  mapping.mapping = getRBFMapping<RBFBackend::PETSc>(_rbfConfig.basisFunction, constraintValue, mapping.fromMesh->getDimensions(), _rbfConfig.supportRadius, _rbfConfig.shapeParameter, _rbfConfig.deadAxis, _rbfConfig.solverRtol, _rbfConfig.polynomial);
591 #else
592  PRECICE_CHECK(false, "The global-iterative RBF solver on a CPU requires a preCICE build with PETSc enabled.");
593 #endif
594  } else if (_rbfConfig.solver == RBFConfiguration::SystemSolver::PUMDirect) {
595  mapping.mapping = getRBFMapping<RBFBackend::PUM>(_rbfConfig.basisFunction, constraintValue, mapping.fromMesh->getDimensions(), _rbfConfig.supportRadius, _rbfConfig.shapeParameter, _rbfConfig.polynomial, _rbfConfig.verticesPerCluster, _rbfConfig.relativeOverlap, _rbfConfig.projectToInput);
596  } else {
597  PRECICE_UNREACHABLE("Unknown RBF solver.");
598  }
599  // 2. any other executor is configured via Ginkgo
600  } else {
601 #ifndef PRECICE_NO_GINKGO
602  _ginkgoParameter = GinkgoParameter();
603  _ginkgoParameter.usePreconditioner = false;
604  _ginkgoParameter.deviceId = _executorConfig->deviceId;
605  if (_executorConfig->executor == ExecutorConfiguration::Executor::CUDA) {
606  _ginkgoParameter.executor = "cuda-executor";
607 #ifndef PRECICE_WITH_CUDA
608  PRECICE_CHECK(false, "The cuda-executor (configured for the mapping from mesh {} to mesh {}) requires a Ginkgo and preCICE build with Cuda enabled.", mapping.fromMesh->getName(), mapping.toMesh->getName());
609 #endif
610  } else if (_executorConfig->executor == ExecutorConfiguration::Executor::HIP) {
611  _ginkgoParameter.executor = "hip-executor";
612 #ifndef PRECICE_WITH_HIP
613  PRECICE_CHECK(false, "The hip-executor (configured for the mapping from mesh {} to mesh {}) requires a Ginkgo and preCICE build with HIP enabled.", mapping.fromMesh->getName(), mapping.toMesh->getName());
614 #endif
615  } else if (_executorConfig->executor == ExecutorConfiguration::Executor::OpenMP) {
616  _ginkgoParameter.executor = "omp-executor";
617  _ginkgoParameter.nThreads = _executorConfig->nThreads;
618 #ifndef PRECICE_WITH_OMP
619  PRECICE_CHECK(false, "The omp-executor (configured for the mapping from mesh {} to mesh {}) requires a Ginkgo and preCICE build with OpenMP enabled.", mapping.fromMesh->getName(), mapping.toMesh->getName());
620 #endif
621  }
622  if (_rbfConfig.solver == RBFConfiguration::SystemSolver::GlobalDirect) {
623  _ginkgoParameter.solver = "qr-solver";
624  } else if (_rbfConfig.solver == RBFConfiguration::SystemSolver::GlobalIterative) {
625  _ginkgoParameter.solver = "cg-solver";
626  _ginkgoParameter.residualNorm = _rbfConfig.solverRtol;
627  } else {
628  PRECICE_UNREACHABLE("Unknown solver type.");
629  }
630  mapping.mapping = getRBFMapping<RBFBackend::Ginkgo>(_rbfConfig.basisFunction, constraintValue, mapping.fromMesh->getDimensions(), _rbfConfig.supportRadius, _rbfConfig.shapeParameter, _rbfConfig.deadAxis, _rbfConfig.polynomial, _ginkgoParameter);
631 #else
632  PRECICE_CHECK(false, "The selected executor for the mapping from mesh {} to mesh {} requires a preCICE build with Ginkgo enabled.", mapping.fromMesh->getName(), mapping.toMesh->getName());
633 #endif
634  }
635 }
