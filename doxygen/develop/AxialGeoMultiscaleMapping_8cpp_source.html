24void AxialGeoMultiscaleMapping::computeMapping()
25{
26 PRECICE_TRACE(output()->nVertices());
27
28 PRECICE_ASSERT(input().get() != nullptr);
29 PRECICE_ASSERT(output().get() != nullptr);
30
31 if (getConstraint() == CONSISTENT) {
32 PRECICE_DEBUG("Compute consistent mapping");
33 if (_type == MultiscaleType::SPREAD) {
34 PRECICE_CHECK(input()->nVertices() == 1, "You can only define an axial geometric multiscale mapping of type spread from a mesh with exactly one vertex.");
35
36 /* When we add support for 1D meshes (https://github.com/precice/precice/issues/1669),
37 we should check for valid dimensions combination.
38
39 const int inDataDimensions = input()->getDimensions();
40 const int outDataDimensions = output()->getDimensions();
41
42 PRECICE_CHECK(input()->getDimensions() == 1, "The input mesh on an axial geometric multiscale mapping can only be 1D at the moment, but it was defined to be {}.", input()->getDimensions())
43 PRECICE_CHECK(output()->getDimensions() == 3, "The output mesh on an axial geometric multiscale mapping can only be 3D at the moment, but it was defined to be {}.", input()->getDimensions())
44 */
45 const int outDataDimensions = 3;
46
47 int effectiveCoordinate = static_cast<std::underlying_type_t<MultiscaleType>>(_axis); // Convert enum struct to int
48 PRECICE_ASSERT(effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::X) ||
49 effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::Y) ||
50 effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::Z),
51 "Unknown multiscale axis type.");
52
53 // compute distances between 1D vertex and 3D vertices
54 mesh::Vertex &v0 = input()->vertex(0);
55 size_t const outSize = output()->nVertices();
56 constexpr double distance_to_radius_threshold = 1.05;
57
58 _vertexDistances.clear();
59 _vertexDistances.reserve(output()->nVertices());
60
61 for (size_t i = 0; i < outSize; i++) {
62 Eigen::VectorXd difference(outDataDimensions);
63 difference = v0.getCoords();
64 difference -= output()->vertex(i).getCoords();
65 double distance_to_radius = difference.norm() / _radius;
66 PRECICE_CHECK(distance_to_radius <= distance_to_radius_threshold, "Output mesh has vertices that do not coincide with the geometric multiscale interface defined by the input mesh. Ratio of vertex distance to radius is {} (which is larger than the assumed threshold of distance_to_radius_threshold).", distance_to_radius);
67 _vertexDistances.push_back(distance_to_radius);
68 }
69 } else {
70 PRECICE_ASSERT(_type == MultiscaleType::COLLECT);
71 PRECICE_CHECK(output()->nVertices() == 1, "You can only define an axial geometric multiscale mapping of type spread from a mesh with exactly one vertex.");
72 // Nothing to do here: A consistent collect mapping only averages all the values, independently of their locations, and this is done in the mapConsistent() method.
73 }
74 } else {
75 PRECICE_ASSERT(getConstraint() == CONSERVATIVE);
76 PRECICE_UNREACHABLE("Axial conservative geometric multiscale mapping is not implemented");
77 PRECICE_DEBUG("Compute conservative mapping");
78 }
79 _hasComputedMapping = true;
80}

96void AxialGeoMultiscaleMapping::mapConsistent(const time::Sample &inData, Eigen::VectorXd &outData)
97{
98 PRECICE_TRACE();
99
100 const int inDataDimensions = inData.dataDims;
101 const Eigen::VectorXd &inputValues = inData.values;
102 Eigen::VectorXd &outputValues = outData;
103 // TODO: check if this needs to change when access to mesh dimension is possible
104 const int outDataDimensions = outData.size() / output()->nVertices();
105
106 // Check that the number of values for the input and output is right according to their dimensions
107 PRECICE_ASSERT((inputValues.size() / static_cast<std::size_t>(inDataDimensions) == input()->nVertices()),
108 inputValues.size(), inDataDimensions, input()->nVertices());
109 PRECICE_ASSERT((outputValues.size() / static_cast<std::size_t>(outDataDimensions) == output()->nVertices()),
110 outputValues.size(), outDataDimensions, output()->nVertices());
111
112 // We currently don't support 1D data, so we need that the user specifies data of the same dimensions on both sides
113 PRECICE_ASSERT(inDataDimensions == outDataDimensions);
114
115 // Effective component (axis) to read/write: 0 for scalar fields (to avoid out-of-bounds), or 0 (x)/ 1 (y)/ 2 (z) for vectors
116 int effectiveCoordinate;
117 if (inDataDimensions == 1) {
118 effectiveCoordinate = 0;
119 } else {
120 effectiveCoordinate = static_cast<std::underlying_type_t<MultiscaleType>>(_axis);
121 PRECICE_ASSERT(effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::X) ||
122 effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::Y) ||
123 effectiveCoordinate == static_cast<std::underlying_type_t<MultiscaleType>>(MultiscaleAxis::Z),
124 "Unknown multiscale axis type.");
125 }
126
127 PRECICE_DEBUG("Map consistent");
128
129 if (_type == MultiscaleType::SPREAD) {
130
131 PRECICE_ASSERT(input()->nVertices() == 1);
132 size_t const outSize = output()->nVertices();
133
134 for (size_t i = 0; i < outSize; i++) {
135 PRECICE_ASSERT(static_cast<size_t>((i * outDataDimensions) + effectiveCoordinate) < static_cast<size_t>(outputValues.size()), ((i * outDataDimensions) + effectiveCoordinate), outputValues.size());
136 if (_profile == SpreadProfile::PARABOLIC) {
137 // When adding support for 2D, remember that this should be 1.5 * inputValues(effectiveCoordinate) * (1 - (_vertexDistances[i] * _vertexDistances[i]));
138 outputValues((i * outDataDimensions) + effectiveCoordinate) = 2 * inputValues(effectiveCoordinate) * (1 - (_vertexDistances[i] * _vertexDistances[i]));
139 } else if (_profile == SpreadProfile::UNIFORM) {
140 outputValues((i * outDataDimensions) + effectiveCoordinate) = inputValues(effectiveCoordinate);
141 }
142 }
143 } else {
144 PRECICE_ASSERT(_type == MultiscaleType::COLLECT);
145 PRECICE_ASSERT(output()->nVertices() == 1);
146 outputValues(effectiveCoordinate) = 0;
147 size_t const inSize = input()->nVertices();
148 for (size_t i = 0; i < inSize; i++) {
149 PRECICE_ASSERT(static_cast<size_t>((i * inDataDimensions) + effectiveCoordinate) < static_cast<size_t>(inputValues.size()),
150 ((i * inDataDimensions) + effectiveCoordinate), inputValues.size());
151 outputValues(effectiveCoordinate) += inputValues((i * inDataDimensions) + effectiveCoordinate);
152 }
153 outputValues(effectiveCoordinate) = outputValues(effectiveCoordinate) / inSize;
154 }
155}
