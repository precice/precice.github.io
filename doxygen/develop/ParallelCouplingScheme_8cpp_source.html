76{
77 PRECICE_ASSERT(math::equals(getTime(), getWindowEndTime()), getTime(), getWindowEndTime());
78 if (isExplicitCouplingScheme()) {
79 if (doesFirstStep()) { // first participant

96 notifyDataHasBeenReceived();

99 doImplicitStep();

103 sendData(getM2N(), getSendData());
