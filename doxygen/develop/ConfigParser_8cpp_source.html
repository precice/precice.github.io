21std::string decodeXML(std::string xml)
22{
23 static const std::map<std::string, char> escapes{{"&lt;", '<'}, {"&gt;", '>'}, {"&amp;", '&'}, {"&quot;", '"'}, {"&apos;", '\''}};
24 while (true) {
25 bool changes{false};
26 for (const auto &kv : escapes) {
27 auto position = xml.find(kv.first);
28 if (position != std::string::npos) {
29 xml.replace(position, kv.first.length(), 1, kv.second);
30 changes = true;
31 }
32 }
33 if (!changes) {
34 break;
35 }
36 };
37 return xml;
38}

158int ConfigParser::readXmlFile(std::string const &filePath)
159{
160 xmlSAXHandler SAXHandler;
161
162 memset(&SAXHandler, 0, sizeof(xmlSAXHandler));
163
164 SAXHandler.initialized = XML_SAX2_MAGIC;
165 SAXHandler.startElementNs = OnStartElementNs;
166 SAXHandler.endElementNs = OnEndElementNs;
167 SAXHandler.characters = OnCharacters;
168 SAXHandler.serror = OnStructuredErrorFunc;
169 SAXHandler.error = OnErrorFunc;
170 SAXHandler.fatalError = OnFatalErrorFunc;
171
172 std::ifstream ifs(filePath);
173 PRECICE_CHECK(ifs, "XML parser was unable to open configuration file \"{}\"", filePath);
174
175 std::string content{std::istreambuf_iterator<char>(ifs), std::istreambuf_iterator<char>()};
176
177 xmlParserCtxtPtr ctxt = xmlCreatePushParserCtxt(&SAXHandler, static_cast<void *>(this),
178 content.c_str(), content.size(), nullptr);
179
180 xmlParseChunk(ctxt, nullptr, 0, 1);
181 xmlFreeParserCtxt(ctxt);
182 xmlCleanupParser();
183
184 return 0;
185}

211void ConfigParser::connectTags(const ConfigurationContext &context, std::vector<std::shared_ptr<XMLTag>> &DefTags, CTagPtrVec &SubTags)
212{
213 std::unordered_set<std::string> usedTags;
214
215 for (auto &subtag : SubTags) {
216 std::string expectedName = (subtag->m_Prefix.length() ? subtag->m_Prefix + ":" : "") + subtag->m_Name;
217 PRECICE_CHECK(expectedName != "solver-interface",
218 "This configuration contains the tag <solver-interface>, meaning it was created for a preCICE version prior to version 3. "
219 "Are you using the correct version of your simulation case? Has this simulation case been updated to this version of preCICE?");
220 const auto tagPosition = std::find_if(
221 DefTags.begin(),
222 DefTags.end(),
223 [expectedName](const std::shared_ptr<XMLTag> &pTag) {
224 return pTag->_fullName == expectedName;
225 });
226
227 if (tagPosition == DefTags.end()) {
228 // Tag not found
229 auto names = gatherCandidates(DefTags, subtag->m_Prefix);
230
231 auto matches = utils::computeMatches(expectedName, names);
232 if (!matches.empty() && matches.front().distance < 3) {
233 matches.erase(std::remove_if(matches.begin(), matches.end(), [](auto &m) { return m.distance > 2; }), matches.end());
234 std::vector<std::string> stringMatches;
235 std::transform(matches.begin(), matches.end(), std::back_inserter(stringMatches), [](auto &m) { return m.name; });
236 PRECICE_ERROR("The configuration contains an unknown tag <{}>. Did you mean <{}>?", expectedName, fmt::join(stringMatches, ">,<"));
237 } else {
238 PRECICE_ERROR("The configuration contains an unknown tag <{}>. Expected tags are {}.", expectedName, fmt::join(names, ", "));
239 }
240 }
241
242 auto pDefSubTag = *tagPosition;
243 pDefSubTag->resetAttributes();
244
245 if ((pDefSubTag->_occurrence == XMLTag::OCCUR_ONCE) || (pDefSubTag->_occurrence == XMLTag::OCCUR_NOT_OR_ONCE)) {
246 PRECICE_CHECK(usedTags.count(pDefSubTag->_fullName) == 0,
247 "Tag <{}> is not allowed to occur multiple times.", pDefSubTag->_fullName);
248 usedTags.emplace(pDefSubTag->_fullName);
249 }
250
251 pDefSubTag->_configuredNamespaces[pDefSubTag->_namespace] = true;
252 pDefSubTag->readAttributes(subtag->m_aAttributes);
253 pDefSubTag->_listener.xmlTagCallback(context, *pDefSubTag);
254 pDefSubTag->_configured = true;
255
256 connectTags(context, pDefSubTag->_subtags, subtag->m_aSubTags);
257
258 pDefSubTag->areAllSubtagsConfigured();
259 pDefSubTag->_listener.xmlEndTagCallback(context, *pDefSubTag);
260 }
