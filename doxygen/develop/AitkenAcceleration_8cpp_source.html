38void AitkenAcceleration::initialize(const DataMap &cplData)
39{
40 checkDataIDs(cplData);

42 PRECICE_CHECK(!data->exchangeSubsteps(),

46 // Accumulate number of entries

49 // Gather sizes

63void AitkenAcceleration::performAcceleration(
64 DataMap &cplData)
65{
66 PRECICE_TRACE();
67
68 // Compute aitken relaxation factor
69 PRECICE_ASSERT(utils::contained(*_primaryDataIDs.begin(), cplData));
70
71 concatenateCouplingData(cplData, _primaryDataIDs, _values, _oldValues);
72
73 // Compute current residual = values - oldValues
74 Eigen::VectorXd residuals = _values - _oldValues;
75
76 // Compute residual deltas (= residuals - oldResiduals) and store it in _oldResiduals
77 Eigen::VectorXd residualDeltas = residuals - _oldResiduals;
78
79 // We need to update the preconditioner in every iteration
80 if (_primaryDataIDs.size() > 1) {
81 _preconditioner->update(false, _values, residuals);
82 }
83
84 // Select/compute aitken factor depending on current iteration count
85 if (_iterationCounter == 0) {
86 // preconditioner not necessary
87 _aitkenFactor = math::sign(_aitkenFactor) * std::min(_initialRelaxation, std::abs(_aitkenFactor));
88 } else {
89 // If we have more than one data set, we scale the data to get a better approximation
90 // of the Aitken factor
91 if (_primaryDataIDs.size() > 1) {
92 _preconditioner->apply(residualDeltas);
93 _preconditioner->apply(_oldResiduals);
94 }
95 // compute fraction of aitken factor with residuals and residual deltas
96 double nominator = utils::IntraComm::dot(_oldResiduals, residualDeltas);
97 double denominator = utils::IntraComm::dot(residualDeltas, residualDeltas);
98 _aitkenFactor = -_aitkenFactor * (nominator / denominator);
99 }
100
101 PRECICE_DEBUG("AitkenFactor: {}", _aitkenFactor);
102
103 // Perform relaxation with aitken factor
104 applyRelaxation(_aitkenFactor, cplData);
105
106 // Store residuals for next iteration
107 _oldResiduals = std::move(residuals);
108
109 _iterationCounter++;
110}

122void AitkenAcceleration::concatenateCouplingData(
123 const DataMap &cplData, const std::vector<DataID> &dataIDs, Eigen::VectorXd &targetValues, Eigen::VectorXd &targetOldValues) const
124{
125 Eigen::Index offset = 0;
126 for (auto id : dataIDs) {
127 Eigen::Index size = cplData.at(id)->values().size();
128 auto & values = cplData.at(id)->values();
129 const auto & oldValues = cplData.at(id)->previousIteration();
130 PRECICE_ASSERT(targetValues.size() >= offset + size, "Target vector was not initialized.", targetValues.size(), offset + size);
131 PRECICE_ASSERT(targetOldValues.size() >= offset + size, "Target vector was not initialized.");
132 for (Eigen::Index i = 0; i < size; i++) {
133 targetValues(i + offset) = values(i);
134 targetOldValues(i + offset) = oldValues(i);
135 }
136 offset += size;
137 }
138}
