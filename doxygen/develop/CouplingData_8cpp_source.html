102void CouplingData::reinitialize()
103{
104 // TODO port this to subcyling
105
106 // The mesh was reinitialized and new written data will be added later in advance().
107 // Meaning all samples are based on a different mesh.
108 // Without remapping, the best we can do is setting them to zero samples.
109 // We keep the timestamps not to break convergence measures, accelerations, and actions
110 auto zero = time::Sample(_data->getDimensions(), _mesh->nVertices());
111 zero.setZero();
112
113 _data->timeStepsStorage().setAllSamples(zero);
114 _previousTimeStepsStorage.setAllSamples(zero);
115}
