295Eigen::VectorXd Mesh::getOwnedVertexData(const Eigen::VectorXd &values)
296{
297 std::vector<double> ownedDataVector;
298 PRECICE_ASSERT(static_cast<std::size_t>(values.size()) >= nVertices());
299 if (empty()) {
300 return {};
301 }
302 int valueDim = values.size() / nVertices();
303 int index = 0;
304
305 for (const auto &vertex : vertices()) {
306 if (vertex.isOwner()) {
307 for (int dim = 0; dim < valueDim; ++dim) {
308 ownedDataVector.push_back(values[index * valueDim + dim]);
309 }
310 }
311 ++index;
312 }
313 Eigen::Map<Eigen::VectorXd> ownedData(ownedDataVector.data(), ownedDataVector.size());
314
315 return ownedData;

325void Mesh::addMesh(
326 Mesh &deltaMesh)
327{
328 PRECICE_TRACE();
329 PRECICE_ASSERT(_dimensions == deltaMesh.getDimensions());
330
331 boost::container::flat_map<VertexID, Vertex *> vertexMap;
332 vertexMap.reserve(deltaMesh.nVertices());
333 Eigen::VectorXd coords(_dimensions);
334 for (const Vertex &vertex : deltaMesh.vertices()) {
335 coords = vertex.getCoords();
336 Vertex &v = createVertex(coords);
337 v.setGlobalIndex(vertex.getGlobalIndex());
338 if (vertex.isTagged())
339 v.tag();
340 v.setOwner(vertex.isOwner());
341 PRECICE_ASSERT(vertex.getID() >= 0, vertex.getID());
342 vertexMap[vertex.getID()] = &v;
343 }
344
345 // you cannot just take the vertices from the edge and add them,
346 // since you need the vertices from the new mesh
347 // (which may differ in IDs)
348 for (const Edge &edge : deltaMesh.edges()) {
349 VertexID vertexIndex1 = edge.vertex(0).getID();
350 VertexID vertexIndex2 = edge.vertex(1).getID();
351 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
352 (vertexMap.count(vertexIndex2) == 1));
353 createEdge(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2]);
354 }
355
356 for (const Triangle &triangle : deltaMesh.triangles()) {
357 VertexID vertexIndex1 = triangle.vertex(0).getID();
358 VertexID vertexIndex2 = triangle.vertex(1).getID();
359 VertexID vertexIndex3 = triangle.vertex(2).getID();
360 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
361 (vertexMap.count(vertexIndex2) == 1) &&
362 (vertexMap.count(vertexIndex3) == 1));
363 createTriangle(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3]);
364 }
365
366 for (const Tetrahedron &tetra : deltaMesh.tetrahedra()) {
367 VertexID vertexIndex1 = tetra.vertex(0).getID();
368 VertexID vertexIndex2 = tetra.vertex(1).getID();
369 VertexID vertexIndex3 = tetra.vertex(2).getID();
370 VertexID vertexIndex4 = tetra.vertex(3).getID();
371
372 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
373 (vertexMap.count(vertexIndex2) == 1) &&
374 (vertexMap.count(vertexIndex3) == 1) &&
375 (vertexMap.count(vertexIndex4) == 1));
376 createTetrahedron(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3], *vertexMap[vertexIndex4]);
377 }
378 _index.clear();
379}

397void Mesh::removeDuplicates()
398{
399 // Remove duplicate tetrahedra
400 auto tetrahedraCnt = _tetrahedra.size();
401 std::sort(_tetrahedra.begin(), _tetrahedra.end());
402 auto lastTetrahedron = std::unique(_tetrahedra.begin(), _tetrahedra.end());
403 _tetrahedra = TetraContainer{_tetrahedra.begin(), lastTetrahedron};
404
405 // Remove duplicate triangles
406 auto triangleCnt = _triangles.size();
407 std::sort(_triangles.begin(), _triangles.end());
408 auto lastTriangle = std::unique(_triangles.begin(), _triangles.end());
409 _triangles = TriangleContainer{_triangles.begin(), lastTriangle};
410
411 // Remove duplicate edges
412 auto edgeCnt = _edges.size();
413 std::sort(_edges.begin(), _edges.end());
414 auto lastEdge = std::unique(_edges.begin(), _edges.end());
415 _edges = EdgeContainer{_edges.begin(), lastEdge};
416
417 PRECICE_DEBUG("Compression removed {} tetrahedra ({} to {}), {} triangles ({} to {}), and {} edges ({} to {})",
418 tetrahedraCnt - _tetrahedra.size(), tetrahedraCnt, _tetrahedra.size(),
419 triangleCnt - _triangles.size(), triangleCnt, _triangles.size(),
420 edgeCnt - _edges.size(), edgeCnt, _edges.size());
421}

448void Mesh::generateImplictPrimitives()
449{
450 if (_triangles.empty() && _tetrahedra.empty()) {
451 PRECICE_DEBUG("No implicit primitives required");
452 return; // no implicit primitives
453 }
454
455 // count explicit primitives for debug
456 const auto explTriangles = _triangles.size();
457 const auto explEdges = _triangles.size();
458
459 // First handle all explicit tetrahedra
460
461 // Build a set of all explicit triangles
462 using ExisitingTriangle = std::tuple<Vertex *, Vertex *, Vertex *>;
463 std::set<ExisitingTriangle> triangles;
464 for (auto &t : _triangles) {
465 triangles.insert(sortedVertexPtrsFor(t));
466 }
467
468 // Generate all missing implicit triangles of explicit tetrahedra
469 // Update the triangles set used by the implicit edge generation
470 auto createTriangleIfMissing = [&](Vertex *a, Vertex *b, Vertex *c) {
471 if (triangles.count({a, b, c}) == 0) {
472 triangles.emplace(a, b, c);
473 createTriangle(*a, *b, *c);
474 };
475 };
476 for (auto &t : _tetrahedra) {
477 auto [a, b, c, d] = sortedVertexPtrsFor(t);
478 // Make sure these are in the same order as above
479 createTriangleIfMissing(a, b, c);
480 createTriangleIfMissing(a, b, d);
481 createTriangleIfMissing(a, c, d);
482 createTriangleIfMissing(b, c, d);
483 }
484
485 // Second handle all triangles, both explicit and implicit from the tetrahedron phase
486 // Build an set of all explicit triangles
487 using ExisitingEdge = std::tuple<Vertex *, Vertex *>;
488 std::set<ExisitingEdge> edges;
489 for (auto &e : _edges) {
490 edges.emplace(sortedVertexPtrsFor(e));
491 }
492
493 // generate all missing implicit edges of implicit and explicit triangles
494 auto createEdgeIfMissing = [&](Vertex *a, Vertex *b) {
495 if (edges.count({a, b}) == 0) {
496 edges.emplace(a, b);
497 createEdge(*a, *b);
498 };
499 };
500 for (auto &t : _triangles) {
501 auto [a, b, c] = sortedVertexPtrsFor(t);
502 // Make sure these are in the same order as above
503 createEdgeIfMissing(a, b);
504 createEdgeIfMissing(a, c);
505 createEdgeIfMissing(b, c);
506 }
507
508 PRECICE_DEBUG("Generated {} implicit triangles and {} implicit edges",
509 _triangles.size() - explTriangles,
510 _edges.size() - explEdges);
511}

513bool Mesh::operator==(const Mesh &other) const
514{
515 bool equal = true;
516 equal &= _vertices.size() == other._vertices.size() &&
517 std::is_permutation(_vertices.begin(), _vertices.end(), other._vertices.begin());
518 equal &= _edges.size() == other._edges.size() &&
519 std::is_permutation(_edges.begin(), _edges.end(), other._edges.begin());
520 equal &= _triangles.size() == other._triangles.size() &&
521 std::is_permutation(_triangles.begin(), _triangles.end(), other._triangles.begin());
522 return equal;
