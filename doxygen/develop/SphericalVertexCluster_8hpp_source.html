125template <typename RADIAL_BASIS_FUNCTION_T>
126SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::SphericalVertexCluster(
127 mesh::Vertex center,
128 double radius,
129 RADIAL_BASIS_FUNCTION_T function,
130 Polynomial polynomial,
131 mesh::PtrMesh inputMesh,
132 mesh::PtrMesh outputMesh)
133 : _center(center), _radius(radius), _polynomial(polynomial), _weightingFunction(radius)
134{
135 PRECICE_TRACE(_center.getCoords(), _radius);
136 precice::profiling::Event eq("map.pou.computeMapping.queryVertices");
137 // Disable integrated polynomial, as it might cause locally singular matrices
138 PRECICE_ASSERT(_polynomial != Polynomial::ON, "Integrated polynomial is not supported for partition of unity data mappings.");
139
140 // Get vertices to be mapped
141 // Subtract a safety margin to exclude the vertices at the edge
142 auto outIDs = outputMesh->index().getVerticesInsideBox(center, radius - math::NUMERICAL_ZERO_DIFFERENCE);
143 // Constructing the partition when we don't have evaluation points is pointless
144 auto inIDs = inputMesh->index().getVerticesInsideBox(center, radius);
145
146 // Transform the vector to the appropriate boost data structure
147 // The IDs are sorted in the boost flat_set, hence, the function here has N log(N) complexity
148 _inputIDs.insert(inIDs.begin(), inIDs.end());
149 _outputIDs.insert(outIDs.begin(), outIDs.end());
150 eq.stop();
151 // If the cluster is empty, we return immediately
152 if (empty()) {
153 return;
154 }
155
156 PRECICE_DEBUG("SphericalVertexCluster input size: {}", inIDs.size());
157 PRECICE_DEBUG("SphericalVertexCluster output size: {}", outIDs.size());
158
159 // The polynomial system is underdetermined if inIDs.size() < dimension + 1. However, the dynamic adoption of the axis in the RBF solver
160 // disables axis, if necessary. Hence, we don't disable the complete polynomial here for underdetermined systems. The case should anyway
161 // only occur for an almost unreasonable small vertices-per-cluster configuration.
162
163 // Construct the solver. Here, the constructor of the RadialBasisFctSolver computes already the decompositions etc, such that we can mark the
164 // mapping in this cluster as computed (mostly for debugging purpose)
165 std::vector<bool> deadAxis(inputMesh->getDimensions(), false);
166 precice::profiling::Event e("map.pou.computeMapping.rbfSolver");
167 _rbfSolver = RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>{function, *inputMesh.get(), _inputIDs, *outputMesh.get(), _outputIDs, deadAxis, _polynomial};
168 _hasComputedMapping = true;
169}

171template <typename RADIAL_BASIS_FUNCTION_T>
172void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::setNormalizedWeight(double normalizedWeight, VertexID id)
173{
174 PRECICE_ASSERT(_outputIDs.size() > 0);
175 PRECICE_ASSERT(_outputIDs.contains(id), id);
176 PRECICE_ASSERT(normalizedWeight > 0);
177
178 if (_normalizedWeights.size() == 0)
179 _normalizedWeights.resize(_outputIDs.size());
180
181 // The find method of boost flat_set comes with O(log(N)) complexity (the more expensive part here)
182 auto localID = _outputIDs.index_of(_outputIDs.find(id));
183
184 PRECICE_ASSERT(static_cast<Eigen::Index>(localID) < _normalizedWeights.size(), localID, _normalizedWeights.size());
185 _normalizedWeights[localID] = normalizedWeight;
186}

188template <typename RADIAL_BASIS_FUNCTION_T>
189void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::mapConservative(const time::Sample &inData, Eigen::VectorXd &outData) const
190{
191 // First, a few sanity checks. Empty partitions shouldn't be stored at all
192 PRECICE_ASSERT(!empty());
193 PRECICE_ASSERT(_hasComputedMapping);
194 PRECICE_ASSERT(_normalizedWeights.size() == static_cast<Eigen::Index>(_outputIDs.size()));
195
196 // Define an alias for data dimension in order to avoid ambiguity
197 const unsigned int nComponents = inData.dataDims;
198 const auto & localInData = inData.values;
199
200 // TODO: We can probably reduce the temporary allocations here
201 Eigen::VectorXd in(_rbfSolver.getOutputSize());
202
203 // Now we perform the data mapping component-wise
204 for (unsigned int c = 0; c < nComponents; ++c) {
205 // Step 1: extract the relevant input data from the global input data and store
206 // it in a contiguous array, which is required for the RBF solver
207 for (unsigned int i = 0; i < _outputIDs.size(); ++i) {
208 const auto dataIndex = *(_outputIDs.nth(i));
209 PRECICE_ASSERT(dataIndex * nComponents + c < localInData.size(), dataIndex * nComponents + c, localInData.size());
210 PRECICE_ASSERT(_normalizedWeights[i] > 0, _normalizedWeights[i], i);
211 // here, we also directly apply the weighting, i.e., we split the input data
212 in[i] = localInData[dataIndex * nComponents + c] * _normalizedWeights[i];
213 }
214
215 // Step 2: solve the system using a conservative constraint
216 auto result = _rbfSolver.solveConservative(in, _polynomial);
217 PRECICE_ASSERT(result.size() == static_cast<Eigen::Index>(_inputIDs.size()));
218
219 // Step 3: now accumulate the result into our global output data
220 for (unsigned int i = 0; i < _inputIDs.size(); ++i) {
221 const auto dataIndex = *(_inputIDs.nth(i));
222 PRECICE_ASSERT(dataIndex * nComponents + c < outData.size(), dataIndex * nComponents + c, outData.size());
223 outData[dataIndex * nComponents + c] += result(i);
224 }
225 }
226}

228template <typename RADIAL_BASIS_FUNCTION_T>
229void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::mapConsistent(const time::Sample &inData, Eigen::VectorXd &outData) const
230{
231 // First, a few sanity checks. Empty partitions shouldn't be stored at all
232 PRECICE_ASSERT(!empty());
233 PRECICE_ASSERT(_hasComputedMapping);
234 PRECICE_ASSERT(_normalizedWeights.size() == static_cast<Eigen::Index>(_outputIDs.size()));
235
236 // Define an alias for data dimension in order to avoid ambiguity
237 const unsigned int nComponents = inData.dataDims;
238 const auto & localInData = inData.values;
239
240 Eigen::VectorXd in(_rbfSolver.getInputSize());
241
242 // Now we perform the data mapping component-wise
243 for (unsigned int c = 0; c < nComponents; ++c) {
244 // Step 1: extract the relevant input data from the global input data and store
245 // it in a contiguous array, which is required for the RBF solver (last polyparams entries remain zero)
246 for (unsigned int i = 0; i < _inputIDs.size(); i++) {
247 const auto dataIndex = *(_inputIDs.nth(i));
248 PRECICE_ASSERT(dataIndex * nComponents + c < localInData.size(), dataIndex * nComponents + c, localInData.size());
249 in[i] = localInData[dataIndex * nComponents + c];
250 }
251
252 // Step 2: solve the system using a consistent constraint
253 auto result = _rbfSolver.solveConsistent(in, _polynomial);
254 PRECICE_ASSERT(static_cast<Eigen::Index>(_outputIDs.size()) == result.size());
255
256 // Step 3: now accumulate the result into our global output data
257 for (unsigned int i = 0; i < _outputIDs.size(); ++i) {
258 const auto dataIndex = *(_outputIDs.nth(i));
259 PRECICE_ASSERT(dataIndex * nComponents + c < outData.size(), dataIndex * nComponents + c, outData.size());
260 PRECICE_ASSERT(_normalizedWeights[i] > 0);
261 // here, we also directly apply the weighting, i.e., split the result data
262 outData[dataIndex * nComponents + c] += result(i) * _normalizedWeights[i];
263 }
264 }

269{
270 // Assume that the local interpolant and the weighting function are the same
