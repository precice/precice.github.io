105 // To discuss: maybe check this somewhere in the ParticipantImpl, as we have now a similar check for the parallel case
106 PRECICE_CHECK(!_bb.empty(), "You are running this participant in serial mode and the bounding box on mesh \"{}\", is empty. Did you call setMeshAccessRegion with valid data?", _mesh->getName());

109 ++nFilteredVertices;

113 "{} vertices on mesh \"{}\" have been filtered out due to the defined bounding box in \"setMeshAccessRegion\" "
114 "in serial mode. Associated data values of the filtered vertices will be filled with zero values in order to provide valid data for other participants when reading data.",
115 nFilteredVertices, _mesh->getName());

117 _mesh->clear();
118 _mesh->addMesh(filteredMesh);

121 mesh::Mesh::VertexDistribution vertexDistribution;
122 int vertexCounter = 0;

124 v.setOwner(true);

137 PRECICE_CHECK(hasAnyMapping() || _allowDirectAccess,
138 "The received mesh {} needs a mapping (either from it, to it, or both) or API access enabled (enable-access=\"true\"). Maybe you don't want to receive this mesh at all?",
139 _mesh->getName());

143 // for RBF-based filtering. https://mediatum.ub.tum.de/doc/1320661/document.pdf
144
145 // (1) Bounding-Box-Filter
146 filterByBoundingBox();

148 // (2) Tag vertices 1st round (i.e. who could be owned by this rank)

158 PRECICE_DEBUG("Tag vertices for filtering: 2nd round.");

164 mesh::Mesh filteredMesh("FilteredMesh", _dimensions, mesh::Mesh::MESH_ID_UNDEFINED);
165 mesh::filterMesh(filteredMesh, *_mesh, [&](const mesh::Vertex &v) { return v.isTagged(); });
166 PRECICE_DEBUG("Mapping filter, filtered from {} to {} vertices, {} to {} edges, and {} to {} triangles.",
167 _mesh->nVertices(), filteredMesh.nVertices(),
168 _mesh->edges().size(), filteredMesh.edges().size(),
169 _mesh->triangles().size(), filteredMesh.triangles().size());
170
171 _mesh->clear();
172 _mesh->addMesh(filteredMesh);
173 e5.stop();
174
175 // (6) Compute vertex distribution or local communication map
176 if (m2n().usesTwoLevelInitialization()) {
177
178 PRECICE_INFO("Compute communication map for mesh {}", _mesh->getName());
179 Event e6("partition.computeCommunicationMap." + _mesh->getName(), profiling::Synchronize);
180
181 // Fill two data structures: remoteCommunicationMap and this rank's communication map (_mesh->getCommunicationMap()).
182 // remoteCommunicationMap: connectedRank -> {remote local vertex index}
183 // _mesh->getCommunicationMap(): connectedRank -> {this rank's local vertex index}
184 // A vertex belongs to a specific connected rank if its global vertex ID lies within the ranks min and max.
185 mesh::Mesh::CommunicationMap remoteCommunicationMap;
186
187 for (size_t vertexIndex = 0; vertexIndex < _mesh->nVertices(); ++vertexIndex) {
188 for (size_t rankIndex = 0; rankIndex < _mesh->getConnectedRanks().size(); ++rankIndex) {
189 int globalVertexIndex = _mesh->vertex(vertexIndex).getGlobalIndex();
190 if (globalVertexIndex <= _remoteMaxGlobalVertexIDs[rankIndex] && globalVertexIndex >= _remoteMinGlobalVertexIDs[rankIndex]) {
191 int remoteRank = _mesh->getConnectedRanks()[rankIndex];
192 remoteCommunicationMap[remoteRank].push_back(globalVertexIndex - _remoteMinGlobalVertexIDs[rankIndex]); // remote local vertex index
193 _mesh->getCommunicationMap()[remoteRank].push_back(vertexIndex); // this rank's local vertex index
194 }
195 }
196 }
197
198 // communicate remote communication map to all remote connected ranks
199 m2n().scatterAllCommunicationMap(remoteCommunicationMap, *_mesh);
200
201 } else {
202

204 Event e6("partition.feedbackMesh." + _mesh->getName(), profiling::Synchronize);
205 if (utils::IntraComm::isSecondary()) {
206 int numberOfVertices = _mesh->nVertices();

208 for (int i = 0; i < numberOfVertices; i++) {
209 vertexIDs[i] = _mesh->vertex(i).getGlobalIndex();
210 }
211 PRECICE_DEBUG("Send partition feedback to primary rank");
212 utils::IntraComm::getCommunication()->sendRange(vertexIDs, 0);
213 } else { // Primary
214 mesh::Mesh::VertexDistribution vertexDistribution;
215 int numberOfVertices = _mesh->nVertices();
216 std::vector<VertexID> vertexIDs(numberOfVertices, -1);
217 for (int i = 0; i < numberOfVertices; i++) {
218 vertexIDs[i] = _mesh->vertex(i).getGlobalIndex();
219 }
220 vertexDistribution[0] = std::move(vertexIDs);
221
222 for (int secondaryRank : utils::IntraComm::allSecondaryRanks()) {
223 PRECICE_DEBUG("Receive partition feedback from the secondary rank {}", secondaryRank);
224 vertexDistribution[secondaryRank] = utils::IntraComm::getCommunication()->receiveRange(secondaryRank, com::asVector<VertexID>);
225 }
226 PRECICE_ASSERT(_mesh->getVertexDistribution().empty());
227 _mesh->setVertexDistribution(std::move(vertexDistribution));
228 }
229 }
230
231 // (7) Compute vertex offsets
232 PRECICE_DEBUG("Compute vertex offsets");
233 if (utils::IntraComm::isSecondary()) {
234
235 // send number of vertices
236 PRECICE_DEBUG("Send number of vertices: {}", _mesh->nVertices());
237 int numberOfVertices = _mesh->nVertices();
238 utils::IntraComm::getCommunication()->send(numberOfVertices, 0);
239
240 // receive vertex offsets
241 mesh::Mesh::VertexOffsets vertexOffsets;
242 utils::IntraComm::getCommunication()->broadcast(vertexOffsets, 0);
243 PRECICE_DEBUG("My vertex offsets: {}", vertexOffsets);
244 PRECICE_ASSERT(_mesh->getVertexOffsets().empty());
245 _mesh->setVertexOffsets(std::move(vertexOffsets));
246
247 } else if (utils::IntraComm::isPrimary()) {
248
249 mesh::Mesh::VertexOffsets vertexOffsets(utils::IntraComm::getSize());
250 vertexOffsets[0] = _mesh->nVertices();
251
252 // receive number of secondary vertices and fill vertex offsets
253 for (int secondaryRank : utils::IntraComm::allSecondaryRanks()) {
254 int numberOfSecondaryRankVertices = -1;
255 utils::IntraComm::getCommunication()->receive(numberOfSecondaryRankVertices, secondaryRank);
256 PRECICE_ASSERT(numberOfSecondaryRankVertices >= 0);

258 }
259

261 PRECICE_DEBUG("My vertex offsets: {}", vertexOffsets);
262 utils::IntraComm::getCommunication()->broadcast(vertexOffsets);
263 PRECICE_ASSERT(_mesh->getVertexOffsets().empty());
264 _mesh->setVertexOffsets(std::move(vertexOffsets));
265 }
266
267 PRECICE_ASSERT(!_mesh->getVertexOffsets().empty());

289void ReceivedPartition::filterByBoundingBox()
290{
291 PRECICE_TRACE(static_cast<int>(_geometricFilter));
292
293 if (m2n().usesTwoLevelInitialization()) {
294 std::string msg = "The received mesh " + _mesh->getName() +
295 " cannot solely be filtered on the primary rank "
296 "(option \"filter-on-primary-rank\") if it is communicated by an m2n communication that uses "
297 "two-level initialization. Use \"filter-on-secondary-rank\" or \"no-filter\" instead.";
298 PRECICE_CHECK(_geometricFilter != ON_PRIMARY_RANK, msg);
299 }
300
301 prepareBoundingBox();
302
303 if (_geometricFilter == ON_PRIMARY_RANK) { // filter on primary rank and communicate reduced mesh then
304
305 PRECICE_ASSERT(not m2n().usesTwoLevelInitialization());
306 PRECICE_INFO("Pre-filter mesh {} by bounding box on primary rank", _mesh->getName());
307 Event e("partition.preFilterMesh." + _mesh->getName(), profiling::Synchronize);
308
309 if (utils::IntraComm::isSecondary()) {
310 PRECICE_DEBUG("Send bounding box to primary rank");
311 com::sendBoundingBox(*utils::IntraComm::getCommunication(), 0, _bb);
312 PRECICE_DEBUG("Receive filtered mesh");
313 com::receiveMesh(*utils::IntraComm::getCommunication(), 0, *_mesh);
314
315 if (isAnyProvidedMeshNonEmpty()) {
316 PRECICE_CHECK(not _mesh->empty(), errorMeshFilteredOut(_mesh->getName(), utils::IntraComm::getRank()));
317 }
318
319 } else { // Primary
320 PRECICE_ASSERT(utils::IntraComm::getRank() == 0);
321 PRECICE_ASSERT(utils::IntraComm::getSize() > 1);
322
323 for (int secondaryRank : utils::IntraComm::allSecondaryRanks()) {
324 mesh::BoundingBox secondaryBB(_bb.getDimension());
325 com::receiveBoundingBox(*utils::IntraComm::getCommunication(), secondaryRank, secondaryBB);
326
327 PRECICE_DEBUG("From secondary rank {}, bounding mesh: {}", secondaryRank, secondaryBB);
328 mesh::Mesh secondaryMesh("SecondaryMesh", _dimensions, mesh::Mesh::MESH_ID_UNDEFINED);
329 mesh::filterMesh(secondaryMesh, *_mesh, [&secondaryBB](const mesh::Vertex &v) { return secondaryBB.contains(v); });
330 PRECICE_DEBUG("Send filtered mesh to secondary rank: {}", secondaryRank);
331 com::sendMesh(*utils::IntraComm::getCommunication(), secondaryRank, secondaryMesh);
332 }
333
334 // Now also filter the remaining primary mesh
335 mesh::Mesh filteredMesh("FilteredMesh", _dimensions, mesh::Mesh::MESH_ID_UNDEFINED);
336 mesh::filterMesh(filteredMesh, *_mesh, [&](const mesh::Vertex &v) { return _bb.contains(v); });
337 PRECICE_DEBUG("Primary rank mesh, filtered from {} to {} vertices, {} to {} edges, and {} to {} triangles.",
338 _mesh->nVertices(), filteredMesh.nVertices(),
339 _mesh->edges().size(), filteredMesh.edges().size(),
340 _mesh->triangles().size(), filteredMesh.triangles().size());
341 _mesh->clear();
342 _mesh->addMesh(filteredMesh);
343
344 if (isAnyProvidedMeshNonEmpty()) {
345 PRECICE_CHECK(not _mesh->empty(), errorMeshFilteredOut(_mesh->getName(), utils::IntraComm::getRank()));
346 }
347 }
348 } else {
349 if (not m2n().usesTwoLevelInitialization()) {
350 PRECICE_INFO("Broadcast mesh {}", _mesh->getName());
351 Event e("partition.broadcastMesh." + _mesh->getName(), profiling::Synchronize);
352
353 if (utils::IntraComm::isSecondary()) {
354 com::broadcastReceiveMesh(*utils::IntraComm::getCommunication(), *_mesh);
355 } else { // Primary
356 PRECICE_ASSERT(utils::IntraComm::isPrimary());
357 com::broadcastSendMesh(*utils::IntraComm::getCommunication(), *_mesh);
358 }
359 }
360 if (_geometricFilter == ON_SECONDARY_RANKS) {
361
362 PRECICE_INFO("Filter mesh {} by bounding box on secondary ranks", _mesh->getName());
363 Event e("partition.filterMeshBB." + _mesh->getName(), profiling::Synchronize);
364
365 mesh::Mesh filteredMesh("FilteredMesh", _dimensions, mesh::Mesh::MESH_ID_UNDEFINED);
366 mesh::filterMesh(filteredMesh, *_mesh, [&](const mesh::Vertex &v) { return _bb.contains(v); });
367
368 PRECICE_DEBUG("Bounding box filter, filtered from {} to {} vertices, {} to {} edges, and {} to {} triangles.",
369 _mesh->nVertices(), filteredMesh.nVertices(),
370 _mesh->edges().size(), filteredMesh.edges().size(),
371 _mesh->triangles().size(), filteredMesh.triangles().size());
372
373 _mesh->clear();
374 _mesh->addMesh(filteredMesh);
375 if (isAnyProvidedMeshNonEmpty()) {
376 PRECICE_CHECK(not _mesh->empty(), errorMeshFilteredOut(_mesh->getName(), utils::IntraComm::getRank()));
377 }
378 } else {
379 PRECICE_ASSERT(_geometricFilter == NO_FILTER);
380 }
381 }

384void ReceivedPartition::compareBoundingBoxes()
385{
386 PRECICE_TRACE();
387
388 _mesh->clear();
389 _mesh->clearPartitioning();
390 _boundingBoxPrepared = false;
391 _remoteMinGlobalVertexIDs.clear();
392 _remoteMaxGlobalVertexIDs.clear();
393
394 // @todo handle coupling mode (i.e. serial participant)
395 // @todo treatment of multiple m2ns
396 PRECICE_ASSERT(_m2ns.size() == 1);
397 if (not m2n().usesTwoLevelInitialization())
398 return;
399
400 // receive and broadcast number of remote ranks
401 int numberOfRemoteRanks = -1;
402 if (utils::IntraComm::isPrimary()) {
403 m2n().getPrimaryRankCommunication()->receive(numberOfRemoteRanks, 0);
404 utils::IntraComm::getCommunication()->broadcast(numberOfRemoteRanks);
405 } else {
406 PRECICE_ASSERT(utils::IntraComm::isSecondary());
407 utils::IntraComm::getCommunication()->broadcast(numberOfRemoteRanks, 0);
408 }
409
410 // define and initialize remote bounding box map
411 mesh::Mesh::BoundingBoxMap remoteBBMap;
412 mesh::BoundingBox initialBB(_mesh->getDimensions());
413
414 for (int remoteRank = 0; remoteRank < numberOfRemoteRanks; remoteRank++) {
415 remoteBBMap.emplace(remoteRank, initialBB);
416 }
417
418 // receive and broadcast remote bounding box map
419 if (utils::IntraComm::isPrimary()) {
420 com::receiveBoundingBoxMap(*m2n().getPrimaryRankCommunication(), 0, remoteBBMap);
421 com::broadcastSendBoundingBoxMap(*utils::IntraComm::getCommunication(), remoteBBMap);
422 } else {
423 PRECICE_ASSERT(utils::IntraComm::isSecondary());
424 com::broadcastReceiveBoundingBoxMap(*utils::IntraComm::getCommunication(), remoteBBMap);
425 }
426
427 // prepare local bounding box
428 prepareBoundingBox();
429
430 if (utils::IntraComm::isPrimary()) { // Primary
431 mesh::Mesh::CommunicationMap connectionMap; // local ranks -> {remote ranks}
432 std::vector<Rank> connectedRanksList; // local ranks with any connection
433
434 // connected ranks for primary rank
435 std::vector<Rank> connectedRanks;
436 for (auto &remoteBB : remoteBBMap) {
437 if (_bb.overlapping(remoteBB.second)) {
438 connectedRanks.push_back(remoteBB.first); // connected remote ranks for this rank
439 }
440 }

442 _mesh->setConnectedRanks(connectedRanks);
443 if (not connectedRanks.empty()) {
444 connectionMap.emplace(0, std::move(connectedRanks));
445 connectedRanksList.push_back(0);
446 }
447

449 for (int rank : utils::IntraComm::allSecondaryRanks()) {
450 std::vector<Rank> secondaryConnectedRanks = utils::IntraComm::getCommunication()->receiveRange(rank, com::asVector<Rank>);
451 if (!secondaryConnectedRanks.empty()) {
452 connectedRanksList.push_back(rank);
453 connectionMap.emplace(rank, std::move(secondaryConnectedRanks));
454 }
455 }
456

458 m2n().getPrimaryRankCommunication()->sendRange(connectedRanksList, 0);

461 "Check that both mapped meshes are describing the same geometry. "
462 "If you deal with very different mesh resolutions, consider increasing the safety-factor in the <receive-mesh /> tag.",

464 com::sendConnectionMap(*m2n().getPrimaryRankCommunication(), 0, connectionMap);
465 } else {
466 PRECICE_ASSERT(utils::IntraComm::isSecondary());
467
468 std::vector<Rank> connectedRanks;
469 for (const auto &remoteBB : remoteBBMap) {
470 if (_bb.overlapping(remoteBB.second)) {
471 connectedRanks.push_back(remoteBB.first);
472 }
473 }
474 PRECICE_ASSERT(_mesh->getConnectedRanks().empty());
475 _mesh->setConnectedRanks(connectedRanks);
476
477 // send connected ranks to primary rank
478 utils::IntraComm::getCommunication()->sendRange(connectedRanks, 0);
479 }

482void ReceivedPartition::prepareBoundingBox()
483{
484 PRECICE_TRACE(_safetyFactor);
485
486 if (_boundingBoxPrepared)
487 return;
488
489 PRECICE_DEBUG("Merge bounding boxes and increase by safety factor");
490
491 // Reset the BoundingBox
492 _bb = mesh::BoundingBox{_dimensions};
493
494 // Create BB around all "other" meshes
495 for (mapping::PtrMapping &fromMapping : _fromMappings) {
496 auto other_bb = fromMapping->getOutputMesh()->getBoundingBox();
497 _bb.expandBy(other_bb);
498 _bb.scaleBy(_safetyFactor);
499 _boundingBoxPrepared = true;
500 }
501 for (mapping::PtrMapping &toMapping : _toMappings) {
502 auto other_bb = toMapping->getInputMesh()->getBoundingBox();
503 _bb.expandBy(other_bb);
504 _bb.scaleBy(_safetyFactor);
505 _boundingBoxPrepared = true;
506 }
507
508 // Expand by user-defined bounding box in case a direct access is desired
509 if (_allowDirectAccess) {
510 auto &other_bb = _mesh->getBoundingBox();
511 _bb.expandBy(other_bb);
512
513 // The safety factor is for mapping based partitionings applied, as usual.
514 // For the direct access, however, we don't apply any safety factor scaling.
515 // If the user defines a safety factor and the partitioning is entirely based
516 // on the defined access region (setMeshAccessRegion), we raise a warning
517 // to inform the user
518 const float defaultSafetyFactor = 0.5;
519 PRECICE_WARN_IF(
520 utils::IntraComm::isPrimary() && !hasAnyMapping() && (_safetyFactor != defaultSafetyFactor),
521 "The received mesh \"{}\" was entirely partitioned based on the defined access region "
522 "(setMeshAccessRegion) and a safety-factor was defined. However, the safety factor "
523 "will be ignored in this case. You may want to modify the access region by modifying "
524 "the specified region in the function itself.",
525 _mesh->getName());
526 _boundingBoxPrepared = true;
527 }

530void ReceivedPartition::createOwnerInformation()
531{
532 PRECICE_TRACE();
533 Event e("partition.createOwnerInformation." + _mesh->getName(), profiling::Synchronize);
534
535 /*
536 We follow different approaches for two-level and one-level methods. For 1LI, a centeralized
537 approach is followed, while the 2LI employs a local/parallel scheme for assigning vertices
538 to corresponding ranks.
539 */
540
541 if (m2n().usesTwoLevelInitialization()) {
542 /*
543 This function ensures that each vertex is owned by only a single rank and
544 is not shared among ranks. Initially, the vertices are checked against the
545 bounding box of each rank. If a vertex fits into only a single bounding box,
546 the vertex is assigned to that rank. If it fits to various bbs, the rank with the
547 lowest number of vertices gets ownership to keep the load as balanced as
548 possible.
549
550 Following steps are taken:
551
552 1- receive local bb map from primary rank
553 2- filter bb map to keep the connected ranks
554 3- own the vertices that only fit into this rank's bb
555 4- send number of owned vertices and the list of shared vertices to neighbors
556 5- for the remaining vertices: check if we have less vertices -> own it!
557 */
558
559 // #1: receive local bb map from primary rank
560 // Define and initialize localBBMap to save local bbs
561
562 mesh::Mesh::BoundingBoxMap localBBMap;
563 for (Rank rank = 0; rank < utils::IntraComm::getSize(); rank++) {
564 localBBMap.emplace(rank, mesh::BoundingBox(_dimensions));
565 }
566
567 // Define a bb map to save the local connected ranks and respective boundingboxes
568 mesh::Mesh::BoundingBoxMap localConnectedBBMap;
569
570 // store global IDs and list of possible shared vertices
571
572 // Global IDs to be saved in:
573 std::vector<VertexID> sharedVerticesGlobalIDs;
574 // Local IDs to be saved in:
575 std::vector<VertexID> sharedVerticesLocalIDs;
576
577 // store possible shared vertices in a map to communicate with neighbors, map: rank -> vertex_global_id
578 mesh::Mesh::CommunicationMap sharedVerticesSendMap;
579
580 // receive list of possible shared vertices from neighboring ranks
581 mesh::Mesh::CommunicationMap sharedVerticesReceiveMap;
582
583 if (utils::IntraComm::isPrimary()) {
584
585 // Insert bounding box of primary ranks
586 localBBMap.at(0) = _bb;
587
588 // primary rank receives local bb from each secondary rank
589 for (int secondaryRank = 1; secondaryRank < utils::IntraComm::getSize(); secondaryRank++) {
590 com::receiveBoundingBox(*utils::IntraComm::getCommunication(), secondaryRank, localBBMap.at(secondaryRank));
591 }
592
593 // primary rank broadcast localBBMap to all secondary ranks
594 com::broadcastSendBoundingBoxMap(*utils::IntraComm::getCommunication(), localBBMap);
595 } else if (utils::IntraComm::isSecondary()) {
596 // secondary ranks send local bb to primary rank
597 com::sendBoundingBox(*utils::IntraComm::getCommunication(), 0, _bb);
598 // secondary ranks receive localBBMap from primary rank
599 com::broadcastReceiveBoundingBoxMap(*utils::IntraComm::getCommunication(), localBBMap);
600 }
601
602 // #2: filter bb map to keep the connected ranks
603 // remove the own bb from the map since we compare the own bb only with other ranks bb.
604 localBBMap.erase(utils::IntraComm::getRank());
605 // find and store local connected ranks
606 for (const auto &localBB : localBBMap) {
607 if (_bb.overlapping(localBB.second)) {
608 localConnectedBBMap.emplace(localBB.first, localBB.second);
609 }
610 }
611
612 // First, insert all comm partnerts, otherwise we end up in a deadlock further down
613 // when exchanging the vector sizes as vertices might be shared from the other
614 // connected rank(s), although the actual size we request here is zero
615 // i.e., we never tell the other ranks that we don't want any vertices
616 // See also test Integration/Parallel/TestBoundingBoxInitializationEmpty
617 for (auto &neighborRank : localConnectedBBMap)
618 sharedVerticesSendMap[neighborRank.first] = std::vector<VertexID>();
619
620 // #3: check vertices and keep only those that fit into the current rank's bb
621 const int numberOfVertices = _mesh->nVertices();
622 PRECICE_DEBUG("Tag vertices, number of vertices {}", numberOfVertices);
623 std::vector<int> tags(numberOfVertices, 1);
624 std::vector<VertexID> globalIDs(numberOfVertices, -1);
625 int ownedVerticesCount = 0; // number of vertices owned by this rank
626 for (int i = 0; i < numberOfVertices; i++) {
627 globalIDs[i] = _mesh->vertex(i).getGlobalIndex();
628 if (_mesh->vertex(i).isTagged()) {
629 bool vertexIsShared = false;
630 for (const auto &neighborRank : localConnectedBBMap) {
631 if (neighborRank.second.contains(_mesh->vertex(i))) {
632 vertexIsShared = true;
633 sharedVerticesSendMap[neighborRank.first].push_back(globalIDs[i]);
634 }
635 }
636
637 if (not vertexIsShared) {
638 tags[i] = 1;
639 ownedVerticesCount++;
640 } else {
641 sharedVerticesGlobalIDs.push_back(globalIDs[i]);
642 sharedVerticesLocalIDs.push_back(i);
643 }
644 } else {
645 tags[i] = 0;
646 }
647 }
648
649 // #4: Exchange number of already owned vertices with the neighbors
650
651 // to store receive requests.
652 std::vector<com::PtrRequest> vertexNumberRequests;
653
654 // Define and initialize for load balancing
655 std::map<int, int> neighborRanksVertexCount;
656 for (auto &neighborRank : localConnectedBBMap) {
657 neighborRanksVertexCount.emplace(neighborRank.first, 0);
658 }
659
660 // Asynchronous receive number of owned vertices from neighbor ranks
661 for (auto &neighborRank : localConnectedBBMap) {
662 auto request = utils::IntraComm::getCommunication()->aReceive(neighborRanksVertexCount.at(neighborRank.first), neighborRank.first);
663 vertexNumberRequests.push_back(request);
664 }
665
666 // Synchronous send number of owned vertices to neighbor ranks
667 for (auto &neighborRank : localConnectedBBMap) {
668 utils::IntraComm::getCommunication()->send(ownedVerticesCount, neighborRank.first);
669 }
670
671 // wait until all aReceives are complete.
672 for (auto &rqst : vertexNumberRequests) {
673 rqst->wait();
674 }
675
676 // Exchange list of shared vertices with the neighbor
677 // to store send requests.
678 std::vector<com::PtrRequest> vertexListRequests;
679
680 for (auto &receivingRank : sharedVerticesSendMap) {
681 int sendSize = receivingRank.second.size();
682 auto request = utils::IntraComm::getCommunication()->aSend(sendSize, receivingRank.first);
683 vertexListRequests.push_back(request);
684 if (sendSize != 0) {
685 auto request = utils::IntraComm::getCommunication()->aSend(span<const int>{receivingRank.second}, receivingRank.first);
686 vertexListRequests.push_back(request);
687 }
688 }
689
690 for (auto &neighborRank : sharedVerticesSendMap) {
691 int receiveSize = 0;
692 utils::IntraComm::getCommunication()->receive(receiveSize, neighborRank.first);
693 if (receiveSize != 0) {
694 std::vector<int> receivedSharedVertices(receiveSize, -1);
695 utils::IntraComm::getCommunication()->receive(span<int>{receivedSharedVertices}, neighborRank.first);
696 sharedVerticesReceiveMap.insert(std::make_pair(neighborRank.first, receivedSharedVertices));
697 }
698 }
699
700 // wait until all aSends are complete.
701 for (auto &rqst : vertexListRequests) {
702 rqst->wait();
703 }
704
705 // #5: Second round assignment according to the number of owned vertices
706 // In case that a vertex can be shared between two ranks, the rank with lower
707 // vertex count will own the vertex.
708 // If both ranks have same vertex count, the lower rank will own the vertex.
709
710 // To do so, we look at all vertices shared with all neighbors
711 for (auto &sharingRank : sharedVerticesReceiveMap) {
712 // First, check if we would change the ownership at all (by default initialization
713 // above, we would be considered as owner of the shared vertices)
714 // If this is fulfilled, we need to set the ownership to false
715 if ((ownedVerticesCount > neighborRanksVertexCount[sharingRank.first]) ||
716 (ownedVerticesCount == neighborRanksVertexCount[sharingRank.first] && utils::IntraComm::getRank() > sharingRank.first)) {
717 // In such a case, we need to find out the tags we need to switch to 'false', as we don't want
718 // to own them any longer. We compute the intersection of all globalIDs this rank shares with
719 // others and the globalIDs of the neighbor rank we are just considering.
720 // The set_intersection_indices gives us the indices in the sharedVerticesGlobalIDs, which we
721 // can use in the sharedVerticesLocalIDs to get the actual index in the 'tags' vector.
722 std::vector<int> res;
723 precice::utils::set_intersection_indices(sharedVerticesGlobalIDs.begin(), sharedVerticesGlobalIDs.begin(), sharedVerticesGlobalIDs.end(),
724 sharingRank.second.begin(), sharingRank.second.end(), std::back_inserter(res));
725 for (auto r : res)
726 tags[sharedVerticesLocalIDs[r]] = static_cast<int>(false);
727 }
728 }
729
730 setOwnerInformation(tags);
731 PRECICE_DEBUG("{} of {} vertices of mesh {} have been filtered out on rank {} since they have no influence on the mapping.",
732 std::count(tags.begin(), tags.end(), 0), tags.size(), _mesh->getName(), utils::IntraComm::getRank());
733 // end of two-level initialization section
734 } else {
735 if (utils::IntraComm::isSecondary()) {
736 int numberOfVertices = _mesh->nVertices();
737 utils::IntraComm::getCommunication()->send(numberOfVertices, 0);
738
739 if (numberOfVertices != 0) {
740 PRECICE_DEBUG("Tag vertices, number of vertices {}", numberOfVertices);
741 std::vector<int> tags(numberOfVertices, -1);
742 std::vector<VertexID> globalIDs(numberOfVertices, -1);
743 bool atInterface = false;
744 for (int i = 0; i < numberOfVertices; i++) {
745 globalIDs[i] = _mesh->vertex(i).getGlobalIndex();
746 if (_mesh->vertex(i).isTagged()) {
747 tags[i] = 1;
748 atInterface = true;
749 } else {
750 tags[i] = 0;
751 }
752 }
753 PRECICE_DEBUG("My tags: {}", tags);
754 PRECICE_DEBUG("My global IDs: {}", globalIDs);
755 PRECICE_DEBUG("Send tags and global IDs");
756 utils::IntraComm::getCommunication()->sendRange(tags, 0);
757 utils::IntraComm::getCommunication()->sendRange(globalIDs, 0);
758 utils::IntraComm::getCommunication()->send(atInterface, 0);
759
760 PRECICE_DEBUG("Receive owner information");
761 std::vector<VertexID> ownerVec = utils::IntraComm::getCommunication()->receiveRange(0, com::asVector<VertexID>);
762 PRECICE_DEBUG("My owner information: {}", ownerVec);
763 PRECICE_ASSERT(ownerVec.size() == static_cast<std::size_t>(numberOfVertices));
764 setOwnerInformation(ownerVec);
765 }
766 }
767
768 else if (utils::IntraComm::isPrimary()) {
769 // To temporary store which vertices already have an owner
770 std::vector<VertexID> globalOwnerVec(_mesh->getGlobalNumberOfVertices(), 0);
771 // The same per rank
772 std::vector<std::vector<VertexID>> secondaryOwnerVecs(utils::IntraComm::getSize());
773 // Global IDs per rank
774 std::vector<std::vector<VertexID>> secondaryGlobalIDs(utils::IntraComm::getSize());
775 // Tag information per rank
776 std::vector<std::vector<int>> secondaryTags(utils::IntraComm::getSize());
777
778 // Fill primary data
779 PRECICE_DEBUG("Tag vertices of primary rank");
780 bool primaryRankAtInterface = false;
781 secondaryOwnerVecs[0].resize(_mesh->nVertices());
782 secondaryGlobalIDs[0].resize(_mesh->nVertices());
783 secondaryTags[0].resize(_mesh->nVertices());
784 for (size_t i = 0; i < _mesh->nVertices(); i++) {
785 secondaryGlobalIDs[0][i] = _mesh->vertex(i).getGlobalIndex();
786 if (_mesh->vertex(i).isTagged()) {
787 primaryRankAtInterface = true;
788 secondaryTags[0][i] = 1;
789 } else {
790 secondaryTags[0][i] = 0;
791 }
792 }
793 PRECICE_DEBUG("My tags: {}", secondaryTags[0]);
794
795 // receive secondary data
796 Rank ranksAtInterface = 0;
797 if (primaryRankAtInterface)
798 ranksAtInterface++;
799
800 for (Rank rank : utils::IntraComm::allSecondaryRanks()) {
801 int localNumberOfVertices = -1;
802 utils::IntraComm::getCommunication()->receive(localNumberOfVertices, rank);
803 PRECICE_DEBUG("Rank {} has {} vertices.", rank, localNumberOfVertices);
804 secondaryOwnerVecs[rank].resize(localNumberOfVertices, 0);
805
806 if (localNumberOfVertices != 0) {
807 PRECICE_DEBUG("Receive tags from secondary rank {}", rank);
808 secondaryTags[rank] = utils::IntraComm::getCommunication()->receiveRange(rank, com::asVector<int>);
809 secondaryGlobalIDs[rank] = utils::IntraComm::getCommunication()->receiveRange(rank, com::asVector<VertexID>);
810 PRECICE_DEBUG("Rank {} has tags {}", rank, secondaryTags[rank]);
811 PRECICE_DEBUG("Rank {} has global IDs {}", rank, secondaryGlobalIDs[rank]);
812 bool atInterface = false;
813 utils::IntraComm::getCommunication()->receive(atInterface, rank);
814 if (atInterface)
815 ranksAtInterface++;
816 }
817 }
818
819 // Decide upon owners,
820 PRECICE_DEBUG("Decide owners, first round by rough load balancing");
821 // Provide a more descriptive error message if direct access was enabled
822 PRECICE_CHECK(!(ranksAtInterface == 0 && _allowDirectAccess),
823 "After repartitioning of mesh \"{}\" all ranks are empty. "
824 "Please check the dimensions of the provided bounding box "
825 "(in \"setMeshAccessRegion\") and verify that it covers vertices "
826 "in the mesh or check the definition of the provided meshes.",
827 _mesh->getName());
828 PRECICE_ASSERT(ranksAtInterface != 0);
829 int localGuess = _mesh->getGlobalNumberOfVertices() / ranksAtInterface; // Guess for a decent load balancing
830 // First round: every secondary rank gets localGuess vertices
831 for (Rank rank : utils::IntraComm::allRanks()) {
832 int counter = 0;
833 for (size_t i = 0; i < secondaryOwnerVecs[rank].size(); i++) {
834 // Vertex has no owner yet and rank could be owner
835 if (globalOwnerVec[secondaryGlobalIDs[rank][i]] == 0 && secondaryTags[rank][i] == 1) {
836 secondaryOwnerVecs[rank][i] = 1; // Now rank is owner
837 globalOwnerVec[secondaryGlobalIDs[rank][i]] = 1; // Vertex now has owner
838 counter++;
839 if (counter == localGuess)
840 break;
841 }
842 }
843 }
844
845 // Second round: distribute all other vertices in a greedy way
846 PRECICE_DEBUG("Decide owners, second round in greedy way");
847 for (Rank rank : utils::IntraComm::allRanks()) {
848 for (size_t i = 0; i < secondaryOwnerVecs[rank].size(); i++) {
849 if (globalOwnerVec[secondaryGlobalIDs[rank][i]] == 0 && secondaryTags[rank][i] == 1) {
850 secondaryOwnerVecs[rank][i] = 1;
851 globalOwnerVec[secondaryGlobalIDs[rank][i]] = rank + 1;
852 }
853 }
854 }
855
856 // Send information back to secondary ranks
857 for (Rank rank : utils::IntraComm::allSecondaryRanks()) {
858 if (not secondaryTags[rank].empty()) {
859 PRECICE_DEBUG("Send owner information to secondary rank {}", rank);
860 utils::IntraComm::getCommunication()->sendRange(secondaryOwnerVecs[rank], rank);
861 }
862 }
863 // Primary rank data
864 PRECICE_DEBUG("My owner information: {}", secondaryOwnerVecs[0]);
865 setOwnerInformation(secondaryOwnerVecs[0]);
866
867#ifndef NDEBUG
868 for (size_t i = 0; i < globalOwnerVec.size(); i++) {
869 PRECICE_DEBUG_IF(globalOwnerVec[i] == 0,
870 "The Vertex with global index {} of mesh: {} was completely filtered out, since it has no influence on any mapping.",
871 i, _mesh->getName());
872 }
873#endif
874 auto filteredVertices = std::count(globalOwnerVec.begin(), globalOwnerVec.end(), 0);
875 PRECICE_WARN_IF(filteredVertices,
876 "{} of {} vertices of mesh {} have been filtered out since they have no influence on the mapping.{}",
877 filteredVertices, _mesh->getGlobalNumberOfVertices(), _mesh->getName(),
878 _allowDirectAccess ? " Associated data values of the filtered vertices will be filled with zero values in order to "
879 "provide valid data for other participants when reading data."
880 : "");
881 }
882 }

905void ReceivedPartition::tagMeshFirstRound()
906{
907 // We want to have every vertex within the box if we access the mesh directly
908 if (_allowDirectAccess) {
909 // _mesh->getBoundingBox is based on the bounding box of the mesh, which is
910 // - set via the API function (setMeshAccessRegion)
911 // - potentially enlarged with another bounding box if there is another mapping defined
912 // i.e. we have a local mesh involved with a bounding box used to enlarge the original one
913 // concluding: it might be that the boundingBox is not (purely) the one asked for by the user
914 // but using mesh-tagAll() would tag the safety margin. Of course, this only applied for combinations of direct access plus mapping, for pure
915 // direct accesses, there is no safety factor
916
917 auto userDefinedBB = _mesh->getBoundingBox();
918 for (auto &vertex : _mesh->vertices()) {
919 if (userDefinedBB.contains(vertex)) {
920 vertex.tag();
921 }
922 }
923 }
924
925 for (const mapping::PtrMapping &fromMapping : _fromMappings) {
926 fromMapping->tagMeshFirstRound();
927 }
928 for (const mapping::PtrMapping &toMapping : _toMappings) {
929 toMapping->tagMeshFirstRound();
930 }
931}
