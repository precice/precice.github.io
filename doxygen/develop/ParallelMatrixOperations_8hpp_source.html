116 // @brief multiplies matrices based on a cyclic communication and block-wise matrix multiplication
117 template <typename Derived1, typename Derived2>
118 void _multiply_cyclic(
119 Eigen::PlainObjectBase<Derived1> &leftMatrix,
120 Eigen::PlainObjectBase<Derived2> &rightMatrix,
121 Eigen::PlainObjectBase<Derived2> &result,
122 const std::vector<int> & offsets,
123 int p, int q, int r)
124 {
125 PRECICE_TRACE();
126 /*
127 * For multiplication W_til * Z = J
128 * -----------------------------------------------------------------------
129 * p = n_global, q = m, r = n_global_primary
130 *
131 * leftMatrix: local: (n_local x m) global: (n_global x m)
132 * rightMatrix: local: (m x n_local_primary) global: (m x n_global_primary)
133 * result: local: (n_global x n_local_primary) global: (n_global x n_global_primary)
134 * -----------------------------------------------------------------------
135 */
136
137 PRECICE_ASSERT(_needCyclicComm);
138 PRECICE_ASSERT(leftMatrix.cols() == q, leftMatrix.cols(), q);
139 PRECICE_ASSERT(result.rows() == p, result.rows(), p);
140
141 //int nextProc = (utils::IntraComm::getRank() + 1) % utils::IntraComm::getSize();
142 int prevProc = (utils::IntraComm::getRank() - 1 < 0) ? utils::IntraComm::getSize() - 1 : utils::IntraComm::getRank() - 1;
143 int rows_rcv = (prevProc > 0) ? offsets[prevProc + 1] - offsets[prevProc] : offsets[1];
144 //Eigen::MatrixXd leftMatrix_rcv = Eigen::MatrixXd::Zero(rows_rcv, q);
145 Eigen::MatrixXd leftMatrix_rcv(rows_rcv, q);
146
147 com::PtrRequest requestSend;
148 com::PtrRequest requestRcv;
149
150 // initiate asynchronous send operation of leftMatrix (W_til) --> nextProc (this data is needed in cycle 1) dim: n_local x cols
151 if (leftMatrix.size() > 0)
152 requestSend = _cyclicCommRight->aSend(leftMatrix, 0);
153
154 // initiate asynchronous receive operation for leftMatrix (W_til) from previous processor --> W_til dim: rows_rcv x cols
155 if (leftMatrix_rcv.size() > 0)
156 requestRcv = _cyclicCommLeft->aReceive(leftMatrix_rcv, 0);
157
158 // compute diagonal blocks where all data is local and no communication is needed
159 // compute block matrices of J_inv of size (n_til x n_til), n_til = local n
160 Eigen::MatrixXd diagBlock(leftMatrix.rows(), leftMatrix.rows());
161 diagBlock.noalias() = leftMatrix * rightMatrix;
162
163 // set block at corresponding row-index on proc
164 int off = offsets[utils::IntraComm::getRank()];
165 PRECICE_ASSERT(result.cols() == diagBlock.cols(), result.cols(), diagBlock.cols());
166 result.block(off, 0, diagBlock.rows(), diagBlock.cols()) = diagBlock;
167
171 for (int cycle = 1; cycle < utils::IntraComm::getSize(); cycle++) {
172
173 // wait until W_til from previous processor is fully received
174 if (requestSend != NULL)
175 requestSend->wait();
176 if (requestRcv != NULL)
177 requestRcv->wait();
178
179 // leftMatrix (leftMatrix_rcv) is available - needed for local multiplication and hand over to next proc
180 Eigen::MatrixXd leftMatrix_copy(leftMatrix_rcv);
181
182 // initiate async send to hand over leftMatrix (W_til) to the next proc (this data will be needed in the next cycle) dim: n_local x cols
183 if (cycle < utils::IntraComm::getSize() - 1) {
184 if (leftMatrix_copy.size() > 0)
185 requestSend = _cyclicCommRight->aSend(leftMatrix_copy, 0);
186 }
187
188 // compute proc that owned leftMatrix_rcv (Wtil_rcv) at the very beginning for each cycle
189 int sourceProc_nextCycle = (utils::IntraComm::getRank() - (cycle + 1) < 0) ? utils::IntraComm::getSize() + (utils::IntraComm::getRank() - (cycle + 1)) : utils::IntraComm::getRank() - (cycle + 1);
190
191 int sourceProc = (utils::IntraComm::getRank() - cycle < 0) ? utils::IntraComm::getSize() + (utils::IntraComm::getRank() - cycle) : utils::IntraComm::getRank() - cycle;
192
193 int rows_rcv_nextCycle = (sourceProc_nextCycle > 0) ? offsets[sourceProc_nextCycle + 1] - offsets[sourceProc_nextCycle] : offsets[1];
194 rows_rcv = (sourceProc > 0) ? offsets[sourceProc + 1] - offsets[sourceProc] : offsets[1];
195 leftMatrix_rcv = Eigen::MatrixXd::Zero(rows_rcv_nextCycle, q);
196
197 // initiate asynchronous receive operation for leftMatrix (W_til) from previous processor --> W_til (this data is needed in the next cycle)
198 if (cycle < utils::IntraComm::getSize() - 1) {
199 if (leftMatrix_rcv.size() > 0) // only receive data, if data has been sent
200 requestRcv = _cyclicCommLeft->aReceive(leftMatrix_rcv, 0);
201 }
202
203 if (requestSend != NULL)
204 requestSend->wait();
205 // compute block with new local data
206 Eigen::MatrixXd block(rows_rcv, rightMatrix.cols());
207 block.noalias() = leftMatrix_copy * rightMatrix;
208
209 // set block at corresponding index in J_inv
210 // the row-offset of the current block is determined by the proc that sends the part of the W_til matrix
211 // note: the direction and ordering of the cyclic sending operation is chosen s.t. the computed block is
212 // local on the current processor (in J_inv).
213 off = offsets[sourceProc];
214 PRECICE_ASSERT(result.cols() == block.cols(), result.cols(), block.cols());
215 result.block(off, 0, block.rows(), block.cols()) = block;
216 }

219 // @brief multiplies matrices based on a dot-product computation with a rectangular result matrix
220 template <typename Derived1, typename Derived2>
221 void _multiplyNM_dotProduct(
222 Eigen::PlainObjectBase<Derived1> &leftMatrix,
223 Eigen::PlainObjectBase<Derived2> &rightMatrix,
224 Eigen::PlainObjectBase<Derived2> &result,
225 const std::vector<int> & offsets,
226 int p, int q, int r)
227 {
228 PRECICE_TRACE();
229 for (int i = 0; i < leftMatrix.rows(); i++) {
230 Rank rank = 0;
231 // find rank of processor that stores the result
232 // the second while is necessary if processors with no vertices are present
233 // Note: the >'=' here is crucial: In case some procs do not have any vertices,
234 // this while loop continues incrementing rank if entries in offsets are equal, i.e.,
235 // it runs to the next non-empty proc.
236 while (i >= offsets[rank + 1])
237 rank++;
238
239 Eigen::VectorXd lMRow = leftMatrix.row(i);
240
241 for (int j = 0; j < r; j++) {
242
243 Eigen::VectorXd rMCol = rightMatrix.col(j);
244 double res_ij = utils::IntraComm::dot(lMRow, rMCol);
245
246 // find proc that needs to store the result.
247 int local_row;
248 if (utils::IntraComm::getRank() == rank) {
249 local_row = i - offsets[rank];
250 result(local_row, j) = res_ij;
251 }
252 }
253 }

257 template <typename Derived1, typename Derived2>
258 void _multiplyNM_block(
259 Eigen::PlainObjectBase<Derived1> &leftMatrix,
260 Eigen::PlainObjectBase<Derived2> &rightMatrix,
261 Eigen::PlainObjectBase<Derived2> &result,
262 const std::vector<int> & offsets,
263 int p, int q, int r)
264 {
265 PRECICE_TRACE();
266
267 // ensure that both matrices are stored in the same order. Important for reduce function, that adds serialized data.
268 PRECICE_ASSERT(static_cast<int>(leftMatrix.IsRowMajor) == static_cast<int>(rightMatrix.IsRowMajor),
269 static_cast<int>(leftMatrix.IsRowMajor), static_cast<int>(rightMatrix.IsRowMajor));
270
271 // multiply local block (saxpy-based approach)
272 // dimension: (n_global x n_local) * (n_local x m) = (n_global x m)
273 Eigen::MatrixXd block = Eigen::MatrixXd::Zero(p, r);
274 block.noalias() = leftMatrix * rightMatrix;
275
276 // all blocks have size (n_global x m)
277 // Note: if procs have no vertices, the block size remains (n_global x m), however,
278 // it must be initialized with zeros, so zeros are added for those procs)
279
280 // sum up blocks on the primary rank, reduce
281 Eigen::MatrixXd summarizedBlocks = Eigen::MatrixXd::Zero(p, r);
282 utils::IntraComm::reduceSum(block, summarizedBlocks);
283
284 // secondary ranks wait to receive their local result
285 if (utils::IntraComm::isSecondary()) {
286 if (result.size() > 0)
287 utils::IntraComm::getCommunication()->receive(result, 0);
288 }
289
290 // primary rank distributes the sub blocks of the results
291 if (utils::IntraComm::isPrimary()) {
292 // distribute blocks of summarizedBlocks (result of multiplication) to corresponding secondary ranks
293 result = summarizedBlocks.block(0, 0, offsets[1], r);
294
295 for (Rank secondaryRank : utils::IntraComm::allSecondaryRanks()) {
296 int off = offsets[secondaryRank];
297 int send_rows = offsets[secondaryRank + 1] - offsets[secondaryRank];
298
299 if (summarizedBlocks.block(off, 0, send_rows, r).size() > 0) {
300 // necessary to save the matrix-block that is to be sent in a temporary matrix-object
301 // otherwise, the send routine walks over the bounds of the block (matrix structure is still from the entire matrix)
302 Eigen::MatrixXd sendBlock = summarizedBlocks.block(off, 0, send_rows, r);
303 utils::IntraComm::getCommunication()->send(sendBlock, secondaryRank);
304 }
305 }
306 }
