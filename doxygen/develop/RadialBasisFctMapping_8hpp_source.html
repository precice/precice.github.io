319{
320 PRECICE_TRACE();

322

324
325 // Gather input data
326 if (utils::IntraComm::isSecondary()) {
327 // Input data is filtered
328 auto localInDataFiltered = this->input()->getOwnedVertexData(inData.values);
329 int localOutputSize = outData.size();
330
331 // Send data and output size
332 utils::IntraComm::getCommunication()->sendRange(localInDataFiltered, 0);
333 utils::IntraComm::getCommunication()->send(localOutputSize, 0);
334
335 } else { // Primary rank or Serial case
336
337 const int valueDim = inData.dataDims;
338
339 std::vector<double> globalInValues(static_cast<std::size_t>(this->input()->getGlobalNumberOfVertices()) * valueDim, 0.0);
340 std::vector<int> outValuesSize;

342 if (utils::IntraComm::isPrimary()) { // Parallel case
343
344 // Filter input data
345 const auto &localInData = this->input()->getOwnedVertexData(inData.values);
346 std::copy(localInData.data(), localInData.data() + localInData.size(), globalInValues.begin());
347 outValuesSize.push_back(outData.size());
348
349 int inputSizeCounter = localInData.size();
350 int secondaryOutDataSize{0};
351
352 for (Rank rank : utils::IntraComm::allSecondaryRanks()) {
353 std::vector<double> secondaryBuffer = utils::IntraComm::getCommunication()->receiveRange(rank, com::asVector<double>);
354 std::copy(secondaryBuffer.begin(), secondaryBuffer.end(), globalInValues.begin() + inputSizeCounter);
355 inputSizeCounter += secondaryBuffer.size();
356
357 utils::IntraComm::getCommunication()->receive(secondaryOutDataSize, rank);
358 outValuesSize.push_back(secondaryOutDataSize);
359 }
360
361 } else { // Serial case
362 const auto &localInData = inData.values;
363 std::copy(localInData.data(), localInData.data() + localInData.size(), globalInValues.begin());
364 outValuesSize.push_back(outData.size());
365 }
366
367 Eigen::VectorXd in;
368
369 in.resize(_rbfSolver->getInputSize()); // rows == n
370 in.setZero();
371
372 // Construct Eigen vectors

377
378 Eigen::VectorXd out;
379 outputValues.setZero();
380
381 // For every data dimension, perform mapping
382 for (int dim = 0; dim < valueDim; dim++) {
383 // Fill input from input data values (last polyparams entries remain zero)
384 for (int i = 0; i < this->input()->getGlobalNumberOfVertices(); i++) {
385 in[i] = inputValues[i * valueDim + dim];

389

392 outputValues[i * valueDim + dim] = out[i];
393 }
394 }
395
396 outData = Eigen::Map<Eigen::VectorXd>(outputValues.data(), outValuesSize.at(0));
397
398 // Data scattering to secondary ranks
399 int beginPoint = outValuesSize.at(0);
400
401 if (utils::IntraComm::isPrimary()) {
402 for (Rank rank : utils::IntraComm::allSecondaryRanks()) {
403 precice::span<const double> toSend{outputValues.data() + beginPoint, static_cast<size_t>(outValuesSize.at(rank))};
404 utils::IntraComm::getCommunication()->sendRange(toSend, rank);
405 beginPoint += outValuesSize.at(rank);
406 }
407 }
408 }
409 if (utils::IntraComm::isSecondary()) {
410 std::vector<double> receivedValues = utils::IntraComm::getCommunication()->receiveRange(0, com::asVector<double>);
411 outData = Eigen::Map<Eigen::VectorXd>(receivedValues.data(), receivedValues.size());
412 }
