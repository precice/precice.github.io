301Eigen::VectorXd Mesh::getOwnedVertexData(const Eigen::VectorXd &values)
302{
303 std::vector<double> ownedDataVector;
304 PRECICE_ASSERT(static_cast<std::size_t>(values.size()) >= nVertices());
305 if (empty()) {

309 int index = 0;

331void Mesh::addMesh(
332 Mesh &deltaMesh)
333{
334 PRECICE_TRACE();
335 PRECICE_ASSERT(_dimensions == deltaMesh.getDimensions());
336
337 boost::container::flat_map<VertexID, Vertex *> vertexMap;
338 vertexMap.reserve(deltaMesh.nVertices());
339 Eigen::VectorXd coords(_dimensions);
340 for (const Vertex &vertex : deltaMesh.vertices()) {
341 coords = vertex.getCoords();
342 Vertex &v = createVertex(coords);
343 v.setGlobalIndex(vertex.getGlobalIndex());
344 if (vertex.isTagged())
345 v.tag();
346 v.setOwner(vertex.isOwner());
347 PRECICE_ASSERT(vertex.getID() >= 0, vertex.getID());
348 vertexMap[vertex.getID()] = &v;
349 }
350
351 // you cannot just take the vertices from the edge and add them,
352 // since you need the vertices from the new mesh
353 // (which may differ in IDs)
354 for (const Edge &edge : deltaMesh.edges()) {
355 VertexID vertexIndex1 = edge.vertex(0).getID();
356 VertexID vertexIndex2 = edge.vertex(1).getID();
357 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
358 (vertexMap.count(vertexIndex2) == 1));
359 createEdge(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2]);
360 }
361
362 for (const Triangle &triangle : deltaMesh.triangles()) {
363 VertexID vertexIndex1 = triangle.vertex(0).getID();
364 VertexID vertexIndex2 = triangle.vertex(1).getID();
365 VertexID vertexIndex3 = triangle.vertex(2).getID();
366 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
367 (vertexMap.count(vertexIndex2) == 1) &&
368 (vertexMap.count(vertexIndex3) == 1));
369 createTriangle(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3]);
370 }
371
372 for (const Tetrahedron &tetra : deltaMesh.tetrahedra()) {
373 VertexID vertexIndex1 = tetra.vertex(0).getID();
374 VertexID vertexIndex2 = tetra.vertex(1).getID();
375 VertexID vertexIndex3 = tetra.vertex(2).getID();
376 VertexID vertexIndex4 = tetra.vertex(3).getID();
377
378 PRECICE_ASSERT((vertexMap.count(vertexIndex1) == 1) &&
379 (vertexMap.count(vertexIndex2) == 1) &&
380 (vertexMap.count(vertexIndex3) == 1) &&
381 (vertexMap.count(vertexIndex4) == 1));
382 createTetrahedron(*vertexMap[vertexIndex1], *vertexMap[vertexIndex2], *vertexMap[vertexIndex3], *vertexMap[vertexIndex4]);
383 }
384 _index.clear();
385}

403void Mesh::removeDuplicates()
404{
405 // Remove duplicate tetrahedra
406 auto tetrahedraCnt = _tetrahedra.size();
407 std::sort(_tetrahedra.begin(), _tetrahedra.end());
408 auto lastTetrahedron = std::unique(_tetrahedra.begin(), _tetrahedra.end());
409 _tetrahedra = TetraContainer{_tetrahedra.begin(), lastTetrahedron};
410
411 // Remove duplicate triangles
412 auto triangleCnt = _triangles.size();
413 std::sort(_triangles.begin(), _triangles.end());
414 auto lastTriangle = std::unique(_triangles.begin(), _triangles.end());
415 _triangles = TriangleContainer{_triangles.begin(), lastTriangle};
416
417 // Remove duplicate edges
418 auto edgeCnt = _edges.size();
419 std::sort(_edges.begin(), _edges.end());
420 auto lastEdge = std::unique(_edges.begin(), _edges.end());
421 _edges = EdgeContainer{_edges.begin(), lastEdge};
422
423 PRECICE_DEBUG("Compression removed {} tetrahedra ({} to {}), {} triangles ({} to {}), and {} edges ({} to {})",
424 tetrahedraCnt - _tetrahedra.size(), tetrahedraCnt, _tetrahedra.size(),
425 triangleCnt - _triangles.size(), triangleCnt, _triangles.size(),
426 edgeCnt - _edges.size(), edgeCnt, _edges.size());
427}

454void Mesh::generateImplictPrimitives()
455{
456 if (_triangles.empty() && _tetrahedra.empty()) {
457 PRECICE_DEBUG("No implicit primitives required");
458 return; // no implicit primitives
459 }
460
461 // count explicit primitives for debug
462 const auto explTriangles = _triangles.size();
463 const auto explEdges = _triangles.size();
464
465 // First handle all explicit tetrahedra
466
467 // Build a set of all explicit triangles
468 using ExisitingTriangle = std::tuple<Vertex *, Vertex *, Vertex *>;
469 std::set<ExisitingTriangle> triangles;
470 for (auto &t : _triangles) {
471 triangles.insert(sortedVertexPtrsFor(t));
472 }
473
474 // Generate all missing implicit triangles of explicit tetrahedra
475 // Update the triangles set used by the implicit edge generation
476 auto createTriangleIfMissing = [&](Vertex *a, Vertex *b, Vertex *c) {
477 if (triangles.count({a, b, c}) == 0) {
478 triangles.emplace(a, b, c);
479 createTriangle(*a, *b, *c);
480 };
481 };
482 for (auto &t : _tetrahedra) {
483 auto [a, b, c, d] = sortedVertexPtrsFor(t);
484 // Make sure these are in the same order as above
485 createTriangleIfMissing(a, b, c);
486 createTriangleIfMissing(a, b, d);
487 createTriangleIfMissing(a, c, d);
488 createTriangleIfMissing(b, c, d);
489 }
490
491 // Second handle all triangles, both explicit and implicit from the tetrahedron phase
492 // Build an set of all explicit triangles
493 using ExisitingEdge = std::tuple<Vertex *, Vertex *>;
494 std::set<ExisitingEdge> edges;
495 for (auto &e : _edges) {
496 edges.emplace(sortedVertexPtrsFor(e));
497 }
498
499 // generate all missing implicit edges of implicit and explicit triangles
500 auto createEdgeIfMissing = [&](Vertex *a, Vertex *b) {
501 if (edges.count({a, b}) == 0) {
502 edges.emplace(a, b);
503 createEdge(*a, *b);
504 };
505 };
506 for (auto &t : _triangles) {
507 auto [a, b, c] = sortedVertexPtrsFor(t);
508 // Make sure these are in the same order as above
509 createEdgeIfMissing(a, b);
510 createEdgeIfMissing(a, c);
511 createEdgeIfMissing(b, c);
512 }
513
514 PRECICE_DEBUG("Generated {} implicit triangles and {} implicit edges",
515 _triangles.size() - explTriangles,
516 _edges.size() - explEdges);
517}

519bool Mesh::operator==(const Mesh &other) const
520{
521 bool equal = true;
522 equal &= _vertices.size() == other._vertices.size() &&
523 std::is_permutation(_vertices.begin(), _vertices.end(), other._vertices.begin());
524 equal &= _edges.size() == other._edges.size() &&
525 std::is_permutation(_edges.begin(), _edges.end(), other._edges.begin());
526 equal &= _triangles.size() == other._triangles.size() &&
527 std::is_permutation(_triangles.begin(), _triangles.end(), other._triangles.begin());
