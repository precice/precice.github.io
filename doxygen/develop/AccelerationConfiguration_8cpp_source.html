150 if (_config.type == VALUE_IQNIMVJ)

160 std::string meshName = callingTag.getStringAttributeValue(ATTR_MESH);
161 auto success = _uniqueDataAndMeshNames.emplace(dataName, meshName);
162 PRECICE_CHECK(success.second,
163 "You have provided a subtag <data name=\"{}\" mesh=\"{}\"/> more than once in your <acceleration:.../>. "
164 "Please remove the duplicated entry.",
165 dataName, meshName);

167 _meshName = callingTag.getStringAttributeValue(ATTR_MESH);

174 "Data with name \"{0}\" associated to mesh \"{1}\" not found on configuration of acceleration. "
175 "Add \"{0}\" to the \"<mesh name={1}>\" tag, or change the data name in the acceleration scheme.",
176 dataName, _meshName);
177
178 const mesh::PtrMesh &mesh = _meshConfig->getMesh(_meshName);
179 const mesh::PtrData &data = mesh->data(dataName);
180 _config.dataIDs.push_back(data->getID());
181 _config.scalings.insert(std::make_pair(data->getID(), scaling));
182
183 _neededMeshes.push_back(_meshName);
184 } else if (callingTag.getName() == TAG_INIT_RELAX) {
185 _userDefinitions.definedRelaxationFactor = true;
186 _config.relaxationFactor = callingTag.getDoubleAttributeValue(ATTR_VALUE);
187 if (callingTag.hasAttribute(ATTR_ENFORCE)) {
188 _config.forceInitialRelaxation = callingTag.getBooleanAttributeValue(ATTR_ENFORCE);
189 } else {
190 _config.forceInitialRelaxation = false;
191 }
192 } else if (callingTag.getName() == TAG_MAX_USED_ITERATIONS) {
193 _userDefinitions.definedMaxIterationsUsed = true;
194 _config.maxIterationsUsed = callingTag.getIntAttributeValue(ATTR_VALUE);
195 } else if (callingTag.getName() == TAG_TIME_WINDOWS_REUSED) {
196 _userDefinitions.definedTimeWindowsReused = true;
197 _config.timeWindowsReused = callingTag.getIntAttributeValue(ATTR_VALUE);
198 } else if (callingTag.getName() == TAG_FILTER) {
199 _userDefinitions.definedFilter = true;
200 const auto &f = callingTag.getStringAttributeValue(ATTR_TYPE);
201 if (f == VALUE_QR1FILTER) {
202 _config.filter = Acceleration::QR1FILTER;
203 } else if (f == VALUE_QR1_ABSFILTER) {
204 _config.filter = Acceleration::QR1FILTER_ABS;
205 } else if (f == VALUE_QR2FILTER) {
206 _config.filter = Acceleration::QR2FILTER;
207 } else {
208 PRECICE_ASSERT(false);
209 }
210 _config.singularityLimit = callingTag.getDoubleAttributeValue(ATTR_SINGULARITYLIMIT);
211 } else if (callingTag.getName() == TAG_PRECONDITIONER) {
212 _userDefinitions.definedPreconditionerType = true;
213 _config.preconditionerType = callingTag.getStringAttributeValue(ATTR_TYPE);
214 _config.precond_nbNonConstTWindows = callingTag.getIntAttributeValue(ATTR_PRECOND_NONCONST_TIME_WINDOWS);
215 } else if (callingTag.getName() == TAG_IMVJRESTART) {
216 _userDefinitions.defineRestartType = true;
217#ifndef PRECICE_NO_MPI
218 _config.imvjChunkSize = callingTag.getIntAttributeValue(ATTR_IMVJCHUNKSIZE);
219 const auto &f = callingTag.getStringAttributeValue(ATTR_TYPE);
220 PRECICE_CHECK((f == VALUE_NO_RESTART) || (!_config.alwaysBuildJacobian), "IMVJ cannot be in restart mode while parameter always-build-jacobian is set to true. "
221 "Please remove 'always-build-jacobian' from the configuration file or do not run in restart mode.");
222 if (f == VALUE_NO_RESTART) {
223 _config.imvjRestartType = IQNIMVJAcceleration::NO_RESTART;
224 } else if (f == VALUE_ZERO_RESTART) {
225 _config.imvjRestartType = IQNIMVJAcceleration::RS_ZERO;
226 } else if (f == VALUE_LS_RESTART) {
227 _config.imvjRSLS_reusedTimeWindows = callingTag.getIntAttributeValue(ATTR_RSLS_REUSED_TIME_WINDOWS);
228 _config.imvjRestartType = IQNIMVJAcceleration::RS_LS;
229 } else if (f == VALUE_SVD_RESTART) {
230 _config.imvjRSSVD_truncationEps = callingTag.getDoubleAttributeValue(ATTR_RSSVD_TRUNCATIONEPS);
231 _config.imvjRestartType = IQNIMVJAcceleration::RS_SVD;
232 } else if (f == VALUE_SLIDE_RESTART) {
233 _config.imvjRestartType = IQNIMVJAcceleration::RS_SLIDE;
234 } else {
235 _config.imvjChunkSize = 0;
236 PRECICE_ASSERT(false);
237 }
238#else
239 PRECICE_ERROR("Acceleration IQN-IMVJ only works if preCICE is compiled with MPI");
240#endif

244void AccelerationConfiguration::xmlEndTagCallback(
245 const xml::ConfigurationContext &context,
246 xml::XMLTag & callingTag)
247{
248 PRECICE_TRACE(callingTag.getName());
249 if (callingTag.getNamespace() == TAG) {
250
251 //create preconditioner
252 if (callingTag.getName() == VALUE_IQNILS || callingTag.getName() == VALUE_AITKEN) {
253 if (_config.preconditionerType == VALUE_CONSTANT_PRECONDITIONER) {
254 _preconditioner = PtrPreconditioner(new ConstantPreconditioner(_config.scalingFactorsInOrder()));
255 } else if (_config.preconditionerType == VALUE_VALUE_PRECONDITIONER) {
256 _preconditioner = PtrPreconditioner(new ValuePreconditioner(_config.precond_nbNonConstTWindows));
257 } else if (_config.preconditionerType == VALUE_RESIDUAL_PRECONDITIONER) {
258 _preconditioner = PtrPreconditioner(new ResidualPreconditioner(_config.precond_nbNonConstTWindows));
259 } else if (_config.preconditionerType == VALUE_RESIDUAL_SUM_PRECONDITIONER) {
260 _preconditioner = PtrPreconditioner(new ResidualSumPreconditioner(_config.precond_nbNonConstTWindows));
261 } else {
262 // no preconditioner defined
263 _preconditioner = PtrPreconditioner(new ResidualSumPreconditioner(_defaultValuesIQNILS.precond_nbNonConstTWindows));
264 }
265 }
266
267 PRECICE_CHECK(!_acceleration, "You are trying to define multiple acceleration schemes, which is not allowed. Please remove one of them.");
268 if (callingTag.getName() == VALUE_CONSTANT) {
269 _acceleration = PtrAcceleration(
270 new ConstantRelaxationAcceleration(
271 _config.relaxationFactor, _config.dataIDs));
272 } else if (callingTag.getName() == VALUE_AITKEN) {
273 _config.relaxationFactor = (_userDefinitions.definedRelaxationFactor) ? _config.relaxationFactor : _defaultAitkenRelaxationFactor;
274 _acceleration = PtrAcceleration(
275 new AitkenAcceleration(
276 _config.relaxationFactor, _config.dataIDs, _preconditioner));
277 } else if (callingTag.getName() == VALUE_IQNILS) {
278 _config.relaxationFactor = (_userDefinitions.definedRelaxationFactor) ? _config.relaxationFactor : _defaultValuesIQNILS.relaxationFactor;
279 _config.maxIterationsUsed = (_userDefinitions.definedMaxIterationsUsed) ? _config.maxIterationsUsed : _defaultValuesIQNILS.maxIterationsUsed;
280 _config.timeWindowsReused = (_userDefinitions.definedTimeWindowsReused) ? _config.timeWindowsReused : _defaultValuesIQNILS.timeWindowsReused;
281 _config.filter = (_userDefinitions.definedFilter) ? _config.filter : _defaultValuesIQNILS.filter;
282 _config.singularityLimit = (_userDefinitions.definedFilter) ? _config.singularityLimit : _defaultValuesIQNILS.singularityLimit;
283 _acceleration = PtrAcceleration(
284 new IQNILSAcceleration(
285 _config.relaxationFactor,
286 _config.forceInitialRelaxation,
287 _config.maxIterationsUsed,
288 _config.timeWindowsReused,
289 _config.filter, _config.singularityLimit,
290 _config.dataIDs,
291 _preconditioner,
292 _config.reducedTimeGridQN));
293 } else if (callingTag.getName() == VALUE_IQNIMVJ) {
294#ifndef PRECICE_NO_MPI
295 _config.relaxationFactor = (_userDefinitions.definedRelaxationFactor) ? _config.relaxationFactor : _defaultValuesIQNIMVJ.relaxationFactor;
296 _config.maxIterationsUsed = (_userDefinitions.definedMaxIterationsUsed) ? _config.maxIterationsUsed : _defaultValuesIQNIMVJ.maxIterationsUsed;
297 _config.timeWindowsReused = (_userDefinitions.definedTimeWindowsReused) ? _config.timeWindowsReused : _defaultValuesIQNIMVJ.timeWindowsReused;
298 _config.filter = (_userDefinitions.definedFilter) ? _config.filter : _defaultValuesIQNILS.filter;
299 _config.singularityLimit = (_userDefinitions.definedFilter) ? _config.singularityLimit : _defaultValuesIQNILS.singularityLimit;
300 if (!_config.alwaysBuildJacobian) {
301 _config.imvjRestartType = (_userDefinitions.defineRestartType) ? _config.imvjRestartType : _defaultValuesIQNIMVJ.imvjRestartType;
302 _config.imvjChunkSize = (_userDefinitions.defineRestartType) ? _config.imvjChunkSize : _defaultValuesIQNIMVJ.imvjChunkSize;
303 }
304
305 // create preconditioner
306 // if imvj restart-mode is of type RS-SVD, max number of non-const preconditioned time windows is limited by the chunksize
307 // it is separated from the other acceleration methods, since SVD-restart might be chosen as default here
308 if (_config.imvjRestartType == IQNIMVJAcceleration::RS_SVD)
309 if (_config.precond_nbNonConstTWindows > _config.imvjChunkSize)
310 _config.precond_nbNonConstTWindows = _config.imvjChunkSize;
311 if (_config.preconditionerType == VALUE_CONSTANT_PRECONDITIONER) {
312 _preconditioner = PtrPreconditioner(new ConstantPreconditioner(_config.scalingFactorsInOrder()));
313 } else if (_config.preconditionerType == VALUE_VALUE_PRECONDITIONER) {
314 _preconditioner = PtrPreconditioner(new ValuePreconditioner(_config.precond_nbNonConstTWindows));
315 } else if (_config.preconditionerType == VALUE_RESIDUAL_PRECONDITIONER) {
316 _preconditioner = PtrPreconditioner(new ResidualPreconditioner(_config.precond_nbNonConstTWindows));
317 } else if (_config.preconditionerType == VALUE_RESIDUAL_SUM_PRECONDITIONER) {
318 _preconditioner = PtrPreconditioner(new ResidualSumPreconditioner(_config.precond_nbNonConstTWindows));
319 } else {
320 // no preconditioner defined
321 _preconditioner = PtrPreconditioner(new ResidualSumPreconditioner(_defaultValuesIQNILS.precond_nbNonConstTWindows));
322 }
323
324 _acceleration = PtrAcceleration(
325 new IQNIMVJAcceleration(
326 _config.relaxationFactor,
327 _config.forceInitialRelaxation,
328 _config.maxIterationsUsed,
329 _config.timeWindowsReused,
330 _config.filter, _config.singularityLimit,
331 _config.dataIDs,
332 _preconditioner,
333 _config.alwaysBuildJacobian,
334 _config.imvjRestartType,
335 _config.imvjChunkSize,
336 _config.imvjRSLS_reusedTimeWindows,
337 _config.imvjRSSVD_truncationEps,
338 _config.reducedTimeGridQN));
339#else
340 PRECICE_ERROR("Acceleration IQN-IMVJ only works if preCICE is compiled with MPI");
341#endif
342 } else {
343 PRECICE_ASSERT(false);
344 }
345 }
