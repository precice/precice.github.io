182void XMLTag::readAttributes(const std::map<std::string, std::string> &aAttributes)
183{
184 PRECICE_TRACE();
185
186 // Check for unexpected attributes and hints
187 for (const auto &element : aAttributes) {
188 const auto &name = element.first;
189 if (hasAttribute(name)) {
190 continue;
191 }
192
193 // check existing hints
194 if (auto pos = _attributeHints.find(name);
195 pos != _attributeHints.end()) {
196 PRECICE_ERROR("The tag <{}> in the configuration contains the attribute \"{}\". {}", _fullName, name, pos->second);
197 }
198
199 auto expected = getAttributeNames();
200 auto matches = utils::computeMatches(name, expected);
201 if (!matches.empty() && matches.front().distance < 3) {
202 matches.erase(std::remove_if(matches.begin(), matches.end(), [](auto &m) { return m.distance > 2; }), matches.end());
203 std::vector<std::string> stringMatches;
204 std::transform(matches.begin(), matches.end(), std::back_inserter(stringMatches), [](auto &m) { return m.name; });
205 PRECICE_ERROR("The tag <{}> in the configuration contains an unknown attribute \"{}\". Did you mean \"{}\"?", _fullName, name, fmt::join(stringMatches, ", "));
206 }
207 PRECICE_ERROR("The tag <{}> in the configuration contains an unknown attribute \"{}\". Expected attributes are {}.", _fullName, name, fmt::join(expected, ", "));
208 }
209
210 // Read all attributes
211 for (auto &attribute : _attributes) {
212 std::visit(
213 [&aAttributes](auto &attribute) { attribute.readValue(aAttributes); },
214 attribute);
215 }
216}

227void XMLTag::areAllSubtagsConfigured() const
228{
229 for (const auto &tag : _subtags) {
230 std::string ns = tag->_namespace;
231 bool configured = tag->isConfigured();
232
233 bool occurOnce = tag->getOccurrence() == OCCUR_ONCE;
234 bool occurOnceOrMore = tag->getOccurrence() == OCCUR_ONCE_OR_MORE;
235
236 if (not ns.empty()) {
237 auto nsIter = _configuredNamespaces.find(ns);
238 PRECICE_ASSERT(nsIter != _configuredNamespaces.end());
239 configured |= nsIter->second;
240 }
241
242 if ((not configured) && (occurOnce || occurOnceOrMore)) {
243
244 if (tag->getNamespace().empty()) {
245 PRECICE_ERROR("Tag <{}> was not found but is required to occur at least once.", tag->getName());
246 } else {
247 PRECICE_ERROR("Tag <{}:... > was not found but is required to occur at least once.", tag->getNamespace());
248 }
249 }
250 }
