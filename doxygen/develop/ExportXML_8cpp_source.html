33void ExportXML::doExport(int index, double time)
34{
35 PRECICE_TRACE(index, time, _mesh->getName());
36
37 if (!keepExport(index))
38 return;
39
40 const auto &mesh = *_mesh;
41
42 processDataNamesAndDimensions(mesh);
43 if (not _location.empty())
44 std::filesystem::create_directories(_location);
45
46 if (isParallel()) {
47 if (_rank == 0) {
48 writeParallelFile(index, time);
49 }
50 if (!mesh.isPartitionEmpty(_rank)) { // only procs at the coupling interface should write output (for performance reasons)
51 writeSubFile(index, time);
52 }
53 } else {
54 writeSubFile(index, time);
55 }

67void ExportXML::processDataNamesAndDimensions(const mesh::Mesh &mesh)
68{
69 _vectorDataNames.clear();
70 _scalarDataNames.clear();
71 const bool isThreeDim = (mesh.getDimensions() == 3);
72 for (const mesh::PtrData &data : mesh.data()) {
73 int dataDimensions = data->getDimensions();
74 const bool hasGradient = data->hasGradient();
75 PRECICE_ASSERT(dataDimensions >= 1);
76 std::string dataName = data->getName();
77 if (dataDimensions == 1) {
78 _scalarDataNames.push_back(dataName);
79 if (hasGradient) {
80 _vectorDataNames.push_back(dataName + "_gradient");
81 }
82 } else {
83 _vectorDataNames.push_back(dataName);
84 if (hasGradient) {
85 _vectorDataNames.push_back(dataName + "_dx");
86 _vectorDataNames.push_back(dataName + "_dy");
87 if (isThreeDim) {
88 _vectorDataNames.push_back(dataName + "_dz");
89 }
90 }
91 }
92 }

191void ExportXML::exportGradient(const mesh::PtrData data, const int spaceDim, std::ostream &outFile) const
192{
193 const auto & gradients = data->timeStepsStorage().last().sample.gradients;
194 const int dataDimensions = data->getDimensions();
195 std::vector<std::string> suffices;
196 if (dataDimensions == 1) {
197 suffices = {"_gradient"};
198 } else if (spaceDim == 2) {
199 suffices = {"_dx", "_dy"};
200 } else if (spaceDim == 3) {
201 suffices = {"_dx", "_dy", "_dz"};
202 }
203 int counter = 0; // Counter for multicomponent
204 for (const auto &suffix : suffices) {
205 const std::string dataName(data->getName());
206 outFile << " <DataArray type=\"Float64\" Name=\"" << dataName << suffix << "\" NumberOfComponents=\"" << 3;
207 outFile << "\" format=\"ascii\">\n";
208 outFile << " ";
209 for (int i = counter; i < gradients.cols(); i += spaceDim) { // Loop over vertices
210 int j = 0;
211 for (; j < gradients.rows(); j++) { // Loop over components
212 outFile << gradients.coeff(j, i) << " ";
213 }
214 if (j < 3) { // If 2D data add additional zero as third component
215 outFile << "0.0"
216 << " ";
217 }
218 }
219 outFile << '\n'
220 << " </DataArray>\n";
221 counter++; // Increment counter for next component
222 }
223}

225void ExportXML::exportData(
226 std::ostream & outFile,
227 const mesh::Mesh &mesh) const
228{
229 outFile << " <PointData Scalars=\"Rank ";
230 for (const auto &scalarDataName : _scalarDataNames) {
231 outFile << scalarDataName << ' ';
232 }
233 outFile << "\" Vectors=\"";
234 for (const auto &vectorDataName : _vectorDataNames) {
235 outFile << vectorDataName << ' ';
236 }
237 outFile << "\">\n";
238
239 // Export the current rank
240 outFile << " <DataArray type=\"UInt32\" Name=\"Rank\" NumberOfComponents=\"1\" format=\"ascii\">\n";
241 outFile << " ";
242 const auto rank = utils::IntraComm::getRank();
243 for (size_t count = 0; count < mesh.nVertices(); ++count) {
244 outFile << rank << ' ';
245 }
246 outFile << "\n </DataArray>\n";
247
248 for (const mesh::PtrData &data : mesh.data()) { // Plot vertex data
249 if (data->timeStepsStorage().empty()) {
250 if (data->hasGradient()) {
251 data->timeStepsStorage().setSampleAtTime(0, time::Sample(data->getDimensions(), mesh.nVertices(), mesh.getDimensions()).setZero());
252 } else {
253 data->timeStepsStorage().setSampleAtTime(0, time::Sample(data->getDimensions(), mesh.nVertices()).setZero());
254 }
255 }
256 const Eigen::VectorXd &values = data->timeStepsStorage().last().sample.values;
257 int dataDimensions = data->getDimensions();
258 std::string dataName(data->getName());
259 int numberOfComponents = (dataDimensions == 2) ? 3 : dataDimensions;
260 const bool hasGradient = data->hasGradient();
261 outFile << " <DataArray type=\"Float64\" Name=\"" << dataName << "\" NumberOfComponents=\"" << numberOfComponents;
262 outFile << "\" format=\"ascii\">\n";
263 outFile << " ";
264 if (dataDimensions > 1) {
265 Eigen::VectorXd viewTemp(dataDimensions);
266 for (size_t count = 0; count < mesh.nVertices(); count++) {
267 size_t offset = count * dataDimensions;
268 for (int i = 0; i < dataDimensions; i++) {
269 viewTemp[i] = values(offset + i);
270 }
271 for (int i = 0; i < dataDimensions; i++) {
272 outFile << viewTemp[i] << ' ';
273 }
274 if (dataDimensions == 2) {
275 outFile << "0.0" << ' '; // 2D data needs to be 3D for vtk
276 }
277 outFile << ' ';
278 }
279 } else if (dataDimensions == 1) {
280 for (size_t count = 0; count < mesh.nVertices(); count++) {
281 outFile << values(count) << ' ';
282 }
283 }
284 outFile << '\n'
285 << " </DataArray>\n";
286 if (hasGradient) {
287 exportGradient(data, dataDimensions, outFile);
288 }
289 }
290 outFile << " </PointData> \n";
291}
