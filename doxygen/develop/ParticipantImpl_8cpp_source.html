343void ParticipantImpl::advance(
344 double computedTimeStepSize)
345{
346
347 PRECICE_TRACE(computedTimeStepSize);
348
349 // Events for the solver time, stopped when we enter, restarted when we leave advance
350 PRECICE_ASSERT(_solverAdvanceEvent, "The advance event is created in initialize");
351 _solverAdvanceEvent->stop();
352
353 Event e("advance", profiling::Fundamental, profiling::Synchronize);
354 profiling::ScopedEventPrefix sep("advance/");
355
356 PRECICE_CHECK(_state != State::Constructed, "initialize() has to be called before advance().");
357 PRECICE_CHECK(_state != State::Finalized, "advance() cannot be called after finalize().");
358 PRECICE_CHECK(_state == State::Initialized, "initialize() has to be called before advance().");
359 PRECICE_ASSERT(_couplingScheme->isInitialized());
360 PRECICE_CHECK(isCouplingOngoing(), "advance() cannot be called when isCouplingOngoing() returns false.");
361 PRECICE_CHECK(!math::equals(computedTimeStepSize, 0.0), "advance() cannot be called with a time step size of 0.");
362 PRECICE_CHECK(computedTimeStepSize > 0.0, "advance() cannot be called with a negative time step size {}.", computedTimeStepSize);
363 _numberAdvanceCalls++;
364
365#ifndef NDEBUG
366 PRECICE_DEBUG("Synchronize time step size");
367 if (utils::IntraComm::isParallel()) {
368 syncTimestep(computedTimeStepSize);
369 }
370#endif
371
372 // Update the coupling scheme time state. Necessary to get correct remainder.
373 const bool isAtWindowEnd = _couplingScheme->addComputedTime(computedTimeStepSize);
374 const double timeSteppedTo = _couplingScheme->getTime();
375 const auto dataToReceive = _couplingScheme->implicitDataToReceive();
376
377 handleDataBeforeAdvance(isAtWindowEnd, timeSteppedTo);
378
379 advanceCouplingScheme();
380
381 // In clase if an implicit scheme, this may be before timeSteppedTo
382 const double timeAfterAdvance = _couplingScheme->getTime();
383 const bool timeWindowComplete = _couplingScheme->isTimeWindowComplete();
384
385 handleDataAfterAdvance(isAtWindowEnd, timeWindowComplete, timeSteppedTo, timeAfterAdvance, dataToReceive);
386
387 PRECICE_INFO(_couplingScheme->printCouplingState());
388
389 PRECICE_DEBUG("Mapped {} samples in write mappings and {} samples in read mappings",
390 _executedWriteMappings, _executedReadMappings);
391
392 _meshLock.lockAll();
393
394 sep.pop();
395 e.stop();
396 _solverAdvanceEvent->start();
397}

416void ParticipantImpl::handleDataAfterAdvance(bool reachedTimeWindowEnd, bool isTimeWindowComplete, double timeSteppedTo, double timeAfterAdvance, const cplscheme::ImplicitData &receivedData)
417{
418 if (!reachedTimeWindowEnd) {
419 // We are subcycling
420 return;
421 }
422
423 if (isTimeWindowComplete) {
424 // Move to next time window
425 PRECICE_ASSERT(math::greaterEquals(timeAfterAdvance, timeSteppedTo), "We must have stayed or moved forwards in time (min-time-step-size).", timeAfterAdvance, timeSteppedTo);
426
427 // As we move forward, there may now be old samples lying around
428 trimOldDataBefore(_couplingScheme->getTimeWindowStart());
429 } else {
430 // We are iterating
431 PRECICE_ASSERT(math::greater(timeSteppedTo, timeAfterAdvance), "We must have moved back in time!");
432
433 trimSendDataAfter(timeAfterAdvance);
434 }
435
436 if (reachedTimeWindowEnd) {
437 trimReadMappedData(timeAfterAdvance, isTimeWindowComplete, receivedData);
438 mapReadData();
439 performDataActions({action::Action::READ_MAPPING_POST});
440 }
441
442 if (isTimeWindowComplete) {
443 // Reset initial guesses for iterative mappings
444 for (auto &context : _accessor->readDataContexts()) {
445 context.resetInitialGuesses();
446 }
447 for (auto &context : _accessor->writeDataContexts()) {
448 context.resetInitialGuesses();
449 }
450 }
451
452 handleExports(ExportTiming::Advance);
453}

836 std::string_view meshName,
837 VertexID first,
838 VertexID second,
839 VertexID third,
840 VertexID fourth)
841{
842 PRECICE_TRACE(meshName, first,
843 second, third, fourth);
844 PRECICE_REQUIRE_MESH_MODIFY(meshName);
845 MeshContext &context = _accessor->usedMeshContext(meshName);
846 PRECICE_CHECK(context.mesh->getDimensions() == 3, "setMeshQuad is only possible for 3D meshes."
847 " Please set the mesh dimension to 3 in the preCICE configuration file.");
848 if (context.meshRequirement == mapping::Mapping::MeshRequirement::FULL) {
849 PRECICE_ASSERT(context.mesh);
850 mesh::Mesh &mesh = *(context.mesh);
851 using impl::errorInvalidVertexID;
852 PRECICE_CHECK(mesh.isValidVertexID(first), errorInvalidVertexID(first));
853 PRECICE_CHECK(mesh.isValidVertexID(second), errorInvalidVertexID(second));
854 PRECICE_CHECK(mesh.isValidVertexID(third), errorInvalidVertexID(third));
855 PRECICE_CHECK(mesh.isValidVertexID(fourth), errorInvalidVertexID(fourth));
856
857 auto vertexIDs = utils::make_array(first, second, third, fourth);
858 PRECICE_CHECK(utils::unique_elements(vertexIDs), "The four vertex ID's are not unique. Please check that the vertices that form the quad are correct.");
859
860 auto coords = mesh::coordsFor(mesh, vertexIDs);
861 PRECICE_CHECK(utils::unique_elements(coords),
862 "The four vertices that form the quad are not unique. The resulting shape may be a point, line or triangle."
863 "Please check that the adapter sends the four unique vertices that form the quad, or that the mesh on the interface is composed of quads.");
864
865 auto convexity = math::geometry::isConvexQuad(coords);
866 PRECICE_CHECK(convexity.convex, "The given quad is not convex. "
867 "Please check that the adapter send the four correct vertices or that the interface is composed of quads.");
868 auto reordered = utils::reorder_array(convexity.vertexOrder, mesh::vertexPtrsFor(mesh, vertexIDs));
869
870 // Vertices are now in the order: V0-V1-V2-V3-V0.
871 // Use the shortest diagonal to split the quad into 2 triangles.
872 // Vertices are now in V0-V1-V2-V3-V0 order. The new edge, e[4] is either 0-2 or 1-3
873 double distance02 = (reordered[0]->getCoords() - reordered[2]->getCoords()).norm();
874 double distance13 = (reordered[1]->getCoords() - reordered[3]->getCoords()).norm();
875
876 // The new edge, e[4], is the shortest diagonal of the quad
877 if (distance02 <= distance13) {
878 mesh.createTriangle(*reordered[0], *reordered[2], *reordered[1]);
879 mesh.createTriangle(*reordered[0], *reordered[2], *reordered[3]);
880 } else {
881 mesh.createTriangle(*reordered[1], *reordered[3], *reordered[0]);
882 mesh.createTriangle(*reordered[1], *reordered[3], *reordered[2]);
883 }

1331void ParticipantImpl::computePartitions()
1332{
1333 // We need to do this in two loops: First, communicate the mesh and later compute the partition.
1334 // Originally, this was done in one loop. This however gave deadlock if two meshes needed to be communicated cross-wise.
1335 // Both loops need a different sorting
1336
1337 auto &contexts = _accessor->usedMeshContexts();
1338
1339 std::sort(contexts.begin(), contexts.end(),
1340 [](MeshContext const *const lhs, MeshContext const *const rhs) -> bool {
1341 return lhs->mesh->getName() < rhs->mesh->getName();
1342 });
1343
1344 for (MeshContext *meshContext : contexts) {
1345 meshContext->partition->communicate();
1346 }
1347
1348 // for two-level initialization, there is also still communication in partition::compute()
1349 // therefore, we cannot resort here.
1350 // @todo this hacky solution should be removed as part of #633
1351 bool resort = true;
1352 for (auto &m2nPair : _m2ns) {
1353 if (m2nPair.second.m2n->usesTwoLevelInitialization()) {
1354 resort = false;
1355 break;
1356 }
1357 }
1358
1359 if (resort) {
1360 // pull provided meshes up front, to have them ready for the decomposition of the received meshes (for the mappings)
1361 std::stable_partition(contexts.begin(), contexts.end(),
1362 [](MeshContext const *const meshContext) -> bool {
1363 return meshContext->provideMesh;
1364 });
1365 }
1366
1367 for (MeshContext *meshContext : contexts) {
1368 meshContext->partition->compute();
1369 if (not meshContext->provideMesh) { // received mesh can only compute their bounding boxes here
1370 meshContext->mesh->computeBoundingBox();
1371 }
1372
1373 meshContext->mesh->allocateDataValues();
1374
1375 const auto requiredSize = meshContext->mesh->nVertices();
1376 for (auto &context : _accessor->writeDataContexts()) {
1377 if (context.getMeshName() == meshContext->mesh->getName()) {
1378 context.resizeBufferTo(requiredSize);
1379 }
1380 }
1381 }
