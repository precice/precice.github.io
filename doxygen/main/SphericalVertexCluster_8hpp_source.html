126template <typename RADIAL_BASIS_FUNCTION_T>
127SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::SphericalVertexCluster(
128 mesh::Vertex center,
129 double radius,
130 RADIAL_BASIS_FUNCTION_T function,
131 Polynomial polynomial,
132 mesh::PtrMesh inputMesh,
133 mesh::PtrMesh outputMesh)
134 : _center(center), _radius(radius), _polynomial(polynomial), _weightingFunction(radius)
135{
136 PRECICE_TRACE(_center.getCoords(), _radius);
137 // Disable integrated polynomial, as it might cause locally singular matrices
138 PRECICE_ASSERT(_polynomial != Polynomial::ON, "Integrated polynomial is not supported for partition of unity data mappings.")
139
140 // Get vertices to be mapped
141 // Subtract a safety margin to exclude the vertices at the edge
142 auto outIDs = outputMesh->index().getVerticesInsideBox(center, radius - math::NUMERICAL_ZERO_DIFFERENCE);
143 // Constructing the partition when we don't have evaluation points is pointless
144 auto inIDs = inputMesh->index().getVerticesInsideBox(center, radius);
145
146 // Transform the vector to the appropriate boost data structure
147 // The IDs are sorted in the boost flat_set, hence, the function here has N log(N) complexity
148 _inputIDs.insert(inIDs.begin(), inIDs.end());
149 _outputIDs.insert(outIDs.begin(), outIDs.end());
150
151 // If the cluster is empty, we return immediately
152 if (empty()) {
153 return;
154 }
155
156 PRECICE_DEBUG("SphericalVertexCluster input size: {}", inIDs.size());
157 PRECICE_DEBUG("SphericalVertexCluster output size: {}", outIDs.size());
158
159 // The polynomial system is underdetermined if inIDs.size() < dimension + 1. However, the dynamic adoption of the axis in the RBF solver
160 // disables axis, if necessary. Hence, we don't disable the complete polynomial here for underdetermined systems. The case should anyway
161 // only occur for an almost unreasonable small vertices-per-cluster configuration.
162
163 // Construct the solver. Here, the constructor of the RadialBasisFctSolver computes already the decompositions etc, such that we can mark the
164 // mapping in this cluster as computed (mostly for debugging purpose)
165 std::vector<bool> deadAxis(inputMesh->getDimensions(), false);
166 _rbfSolver = RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>{function, *inputMesh.get(), _inputIDs, *outputMesh.get(), _outputIDs, deadAxis, _polynomial};
167 _hasComputedMapping = true;
168}

170template <typename RADIAL_BASIS_FUNCTION_T>
171void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::setNormalizedWeight(double normalizedWeight, VertexID id)
172{
173 PRECICE_ASSERT(_outputIDs.size() > 0);
174 PRECICE_ASSERT(_outputIDs.contains(id), id);
175 PRECICE_ASSERT(normalizedWeight > 0);
176
177 if (_normalizedWeights.size() == 0)
178 _normalizedWeights.resize(_outputIDs.size());
179
180 // The find method of boost flat_set comes with O(log(N)) complexity (the more expensive part here)
181 auto localID = _outputIDs.index_of(_outputIDs.find(id));
182
183 PRECICE_ASSERT(static_cast<Eigen::Index>(localID) < _normalizedWeights.size(), localID, _normalizedWeights.size());
184 _normalizedWeights[localID] = normalizedWeight;
185}

187template <typename RADIAL_BASIS_FUNCTION_T>
188void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::mapConservative(const time::Sample &inData, Eigen::VectorXd &outData) const
189{
190 // First, a few sanity checks. Empty partitions shouldn't be stored at all
191 PRECICE_ASSERT(!empty());
192 PRECICE_ASSERT(_hasComputedMapping);
193 PRECICE_ASSERT(_normalizedWeights.size() == static_cast<Eigen::Index>(_outputIDs.size()));
194
195 // Define an alias for data dimension in order to avoid ambiguity
196 const unsigned int nComponents = inData.dataDims;
197 const auto & localInData = inData.values;
198
199 // TODO: We can probably reduce the temporary allocations here
200 Eigen::VectorXd in(_rbfSolver.getOutputSize());
201
202 // Now we perform the data mapping component-wise
203 for (unsigned int c = 0; c < nComponents; ++c) {
204 // Step 1: extract the relevant input data from the global input data and store
205 // it in a contiguous array, which is required for the RBF solver
206 for (unsigned int i = 0; i < _outputIDs.size(); ++i) {
207 const auto dataIndex = *(_outputIDs.nth(i));
208 PRECICE_ASSERT(dataIndex * nComponents + c < localInData.size(), dataIndex * nComponents + c, localInData.size());
209 PRECICE_ASSERT(_normalizedWeights[i] > 0, _normalizedWeights[i], i);
210 // here, we also directly apply the weighting, i.e., we split the input data
211 in[i] = localInData[dataIndex * nComponents + c] * _normalizedWeights[i];
212 }
213
214 // Step 2: solve the system using a conservative constraint
215 auto result = _rbfSolver.solveConservative(in, _polynomial);
216 PRECICE_ASSERT(result.size() == static_cast<Eigen::Index>(_inputIDs.size()));
217
218 // Step 3: now accumulate the result into our global output data
219 for (unsigned int i = 0; i < _inputIDs.size(); ++i) {
220 const auto dataIndex = *(_inputIDs.nth(i));
221 PRECICE_ASSERT(dataIndex * nComponents + c < outData.size(), dataIndex * nComponents + c, outData.size());
222 outData[dataIndex * nComponents + c] += result(i);
223 }
224 }
225}

227template <typename RADIAL_BASIS_FUNCTION_T>
228void SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::mapConsistent(const time::Sample &inData, Eigen::VectorXd &outData) const
229{
230 // First, a few sanity checks. Empty partitions shouldn't be stored at all
231 PRECICE_ASSERT(!empty());
232 PRECICE_ASSERT(_hasComputedMapping);
233 PRECICE_ASSERT(_normalizedWeights.size() == static_cast<Eigen::Index>(_outputIDs.size()));
234
235 // Define an alias for data dimension in order to avoid ambiguity
236 const unsigned int nComponents = inData.dataDims;
237 const auto & localInData = inData.values;
238
239 Eigen::VectorXd in(_rbfSolver.getInputSize());
240
241 // Now we perform the data mapping component-wise
242 for (unsigned int c = 0; c < nComponents; ++c) {
243 // Step 1: extract the relevant input data from the global input data and store
244 // it in a contiguous array, which is required for the RBF solver (last polyparams entries remain zero)
245 for (unsigned int i = 0; i < _inputIDs.size(); i++) {
246 const auto dataIndex = *(_inputIDs.nth(i));
247 PRECICE_ASSERT(dataIndex * nComponents + c < localInData.size(), dataIndex * nComponents + c, localInData.size());
248 in[i] = localInData[dataIndex * nComponents + c];
249 }
250
251 // Step 2: solve the system using a consistent constraint
252 auto result = _rbfSolver.solveConsistent(in, _polynomial);
253 PRECICE_ASSERT(static_cast<Eigen::Index>(_outputIDs.size()) == result.size());
254
255 // Step 3: now accumulate the result into our global output data
256 for (unsigned int i = 0; i < _outputIDs.size(); ++i) {
257 const auto dataIndex = *(_outputIDs.nth(i));
258 PRECICE_ASSERT(dataIndex * nComponents + c < outData.size(), dataIndex * nComponents + c, outData.size());
259 PRECICE_ASSERT(_normalizedWeights[i] > 0);
260 // here, we also directly apply the weighting, i.e., split the result data
261 outData[dataIndex * nComponents + c] += result(i) * _normalizedWeights[i];
262 }
263 }

267double SphericalVertexCluster<RADIAL_BASIS_FUNCTION_T>::computeWeight(const mesh::Vertex &v) const
268{
269 // Assume that the local interpolant and the weighting function are the same
270 // TODO: We don't need to reduce the dead coordinates here as the values should reduce anyway
271 auto res = computeSquaredDifference(_center.rawCoords(), v.rawCoords(), {{true, true, true}});
272 return _weightingFunction.evaluate(std::sqrt(res));
