69void DataContext::appendMapping(MappingContext mappingContext)
70{
71 PRECICE_ASSERT(mappingContext.fromData);
72 PRECICE_ASSERT(mappingContext.toData);
73 // Make sure we don't append a mapping twice
74#ifndef NDEBUG
75 for (auto &context : _mappingContexts) {
76 PRECICE_ASSERT(!((context.mapping == mappingContext.mapping) && (context.fromData == mappingContext.fromData) && (context.fromData == mappingContext.toData)), "The appended mapping already exists.");
77 }
78#endif
79 _mappingContexts.emplace_back(mappingContext);
80 PRECICE_ASSERT(mappingContext.fromData == _providedData || mappingContext.toData == _providedData, "Either fromData or toData has to equal _providedData.");
81 PRECICE_ASSERT(mappingContext.fromData->getName() == getDataName());
82 PRECICE_ASSERT(mappingContext.toData->getName() == getDataName());
83}

90int DataContext::mapData(std::optional<double> after, bool skipZero)
91{
92 PRECICE_TRACE(getMeshName(), getDataName());
93 PRECICE_ASSERT(hasMapping());
94
95 int executedMappings{0};
96
97 // Execute the mappings
98 for (auto &context : _mappingContexts) {
99 PRECICE_CHECK(!context.fromData->stamples().empty(),
100 "Data {0} on mesh {1} didn't contain any data samples while attempting to map to mesh {2}. "
101 "Check your exchange tags to ensure your coupling scheme exchanges the data or the pariticipant produces it using an action. "
102 "The expected exchange tag should look like this: <exchange data=\"{0}\" mesh=\"{1}\" from=... to=... />.",
103 context.fromData->getName(), context.mapping->getInputMesh()->getName(), context.mapping->getOutputMesh()->getName());
104
105 // linear lookup should be sufficient here
106 const auto timestampExists = [times = context.toData->timeStepsStorage().getTimes()](double lookup) -> bool {
107 return std::any_of(times.data(), std::next(times.data(), times.size()), [lookup](double time) {
108 return math::equals(time, lookup);
109 });
110 };
111
112 auto &mapping = *context.mapping;
113
114 const auto dataDims = context.fromData->getDimensions();
115
116 for (const auto &stample : context.fromData->stamples()) {
117 // skip stamples before given time
118 if (after && math::smallerEquals(stample.timestamp, *after)) {
119 PRECICE_DEBUG("Skipping stample t={} (not after {})", stample.timestamp, *after);
120 continue;
121 }
122 // skip existing stamples
123 if (timestampExists(stample.timestamp)) {
124 PRECICE_DEBUG("Skipping stample t={} (exists)", stample.timestamp);
125 continue;
126 }
127
128 time::Sample outSample{
129 dataDims,
130 Eigen::VectorXd::Zero(dataDims * mapping.getOutputMesh()->nVertices())};
131
132 // Note that the l2norm is only computed during initialization due to short-circuit evaluation in C++
133 bool skipMapping = skipZero && (utils::IntraComm::l2norm(stample.sample.values) < math::NUMERICAL_ZERO_DIFFERENCE);
134
135 PRECICE_INFO("Mapping \"{}\" for t={} from \"{}\" to \"{}\"{}",
136 getDataName(), stample.timestamp, mapping.getInputMesh()->getName(), mapping.getOutputMesh()->getName(),
137 (skipMapping ? " (skipped zero sample)" : ""));
138 if (!skipMapping) {
139 if (mapping.requiresInitialGuess()) {
140 const FromToDataIDs key{context.fromData->getID(), context.toData->getID()};
141 mapping.map(stample.sample, outSample.values, _initialGuesses[key]);
142 } else {
143 mapping.map(stample.sample, outSample.values);
144 }
145 PRECICE_DEBUG("Mapped values (t={}) = {}", stample.timestamp, utils::previewRange(3, outSample.values));
146 ++executedMappings;
147 }
148
149 // Store data from mapping buffer in storage
150 context.toData->setSampleAtTime(stample.timestamp, std::move(outSample));
151 }
152 }
153 return executedMappings;
154}
