283template <typename ElementType, std::size_t Extent>
284class span {
285 static_assert(std::is_object<ElementType>::value,
286 "A span's ElementType must be an object type (not a "
287 "reference type or void)");
288 static_assert(detail::is_complete<ElementType>::value,
289 "A span's ElementType must be a complete type (not a forward "
290 "declaration)");
291 static_assert(!std::is_abstract<ElementType>::value,
292 "A span's ElementType cannot be an abstract class type");

294 using storage_type = detail::span_storage<ElementType, Extent>;

300 using size_type = std::size_t;
301 using difference_type = std::ptrdiff_t;
302 using pointer = element_type*;
303 using const_pointer = const element_type*;
304 using reference = element_type&;

306 using iterator = pointer;

318 PRECICE_SPAN_CONSTEXPR11 span(pointer ptr, size_type count)

324 PRECICE_SPAN_CONSTEXPR11 span(pointer first_elem, pointer last_elem)
