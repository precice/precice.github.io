142void TestContext::initializeMPI(const TestContext::Participants &participants)
143{
144 auto baseComm = Par::current();
145 const int globalRank = baseComm->rank();
146 const int available = baseComm->size();
147
148 // groups contain the accumulated sizes of previous groups
149 std::vector<int> groups(participants.size());
150 std::transform(participants.begin(), participants.end(), groups.begin(), [](const auto &p) { return p.size; });
151 std::partial_sum(groups.begin(), groups.end(), groups.begin());
152
153 // Check if there are enough ranks available
154 auto required = groups.back();
155 if (required > available) {
156 throw std::runtime_error{"This test requests " + std::to_string(required) + " ranks, but there are only " + std::to_string(available) + " available"};
157 }
158
159 // Check if this rank isn't needed
160 if (globalRank >= required) {
161 Par::splitCommunicator(); // No group
162 invalid = true;
163 return;
164 }
165
166 // Find the participant this rank is assigned to
167 auto position = std::upper_bound(groups.begin(), groups.end(), globalRank);
168 auto participant = std::distance(groups.begin(), position);
169 Par::splitCommunicator(participant);
170 setContextFrom(participants[participant]);
171}

233m2n::PtrM2N TestContext::connectPrimaryRanks(const std::string &acceptor, const std::string &requestor, const ConnectionOptions &options) const
234{
235 auto participantCom = com::PtrCommunication(new com::SocketCommunication());
236
237 m2n::DistributedComFactory::SharedPointer distrFactory;
238 switch (options.type) {
239 case ConnectionType::GatherScatter:
240 distrFactory.reset(new m2n::GatherScatterComFactory(participantCom));
241 break;
242 case ConnectionType::PointToPoint:
243 distrFactory.reset(new m2n::PointToPointComFactory(com::PtrCommunicationFactory(new com::SocketCommunicationFactory())));
244 break;
245 default:
246 throw std::runtime_error{"ConnectionType unknown"};
247 };
248 auto m2n = m2n::PtrM2N(new m2n::M2N(participantCom, distrFactory, options.useOnlyPrimaryCom, options.useTwoLevelInit));
249
250 if (std::find(_names.begin(), _names.end(), acceptor) == _names.end()) {
251 throw std::runtime_error{
252 "Acceptor \"" + acceptor + "\" not defined in this context."};
253 }
254 if (std::find(_names.begin(), _names.end(), requestor) == _names.end()) {
255 throw std::runtime_error{
256 "Requestor \"" + requestor + "\" not defined in this context."};
257 }
258
259 if (isNamed(acceptor)) {
260 m2n->acceptPrimaryRankConnection(acceptor, requestor);
261 } else if (isNamed(requestor)) {
262 m2n->requestPrimaryRankConnection(acceptor, requestor);
263 } else {
264 throw std::runtime_error{"You try to connect " + acceptor + " and " + requestor + ", but this context is named " + name};
265 }
266 return m2n;
267}
