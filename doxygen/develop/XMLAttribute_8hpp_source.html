161template <typename ATTRIBUTE_T>
162void XMLAttribute<ATTRIBUTE_T>::readValue(const std::map<std::string, std::string> &aAttributes)
163{
164 PRECICE_TRACE(_name);
165 PRECICE_ASSERT(!_read, "Attribute \"" + _name + "\" has already been read.");
166
167 const auto position = aAttributes.find(getName());
168 if (position == aAttributes.end()) {
169 PRECICE_CHECK(_hasDefaultValue, "Attribute \"{}\" is required, but was not defined.", _name);
170 set(_value, _defaultValue);
171 } else {
172 try {
173 readValueSpecific(position->second, _value);
174 } catch (const std::exception &e) {
175 PRECICE_ERROR(e.what());
176 }
177 if (_hasValidation) {
178 if (std::find(_options.begin(), _options.end(), _value) == _options.end()) {
179 std::ostringstream stream;
180 stream << "Invalid value \"" << _value << "\" of attribute \""
181 << getName() << "\": ";
182 // print first
183 auto first = _options.begin();
184 stream << "value must be \"" << *first << '"';
185 ++first;
186 // print the remaining with separator
187 for (; first != _options.end(); ++first) {
188 stream << " or value must be \"" << *first << '"';
189 }
190
191 PRECICE_ERROR(stream.str());
192 }
193 }
194 }
195 PRECICE_DEBUG("Read valid attribute \"{}\" value = {}", getName(), _value);
196}
