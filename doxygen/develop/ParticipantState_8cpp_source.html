383void ParticipantState::exportIntermediate(IntermediateExport exp)
384{
385 for (const io::ExportContext &context : exportContexts()) {
386 if (context.everyIteration) {
387 PRECICE_DEBUG("Exporting mesh {} for iteration {} to location \"{}\"", context.meshName, exp.iteration, context.location);
388 context.exporter->doExport(exp.iteration, exp.time);
389 continue;
390 }
391 if (exp.complete) {
392 PRECICE_DEBUG("Exporting mesh {} for timewindow {} to location \"{}\"", context.meshName, exp.timewindow, context.location);
393 context.exporter->doExport(exp.timewindow, exp.time);
394 }
395
396 if (exp.final || (exp.complete && context.updateSeries)) {
397 PRECICE_DEBUG("Exporting series file of mesh {} to location \"{}\"", context.meshName, context.location);
398 context.exporter->exportSeries();
399 }
400 }
401
402 if (exp.complete) {
403 // Export watch point data
404 for (const PtrWatchPoint &watchPoint : watchPoints()) {
405 watchPoint->exportPointData(exp.time);
406 }
407
408 for (const PtrWatchIntegral &watchIntegral : watchIntegrals()) {
409 watchIntegral->exportIntegralData(exp.time);
410 }
411 }

432std::string ParticipantState::hintForMesh(std::string_view mesh) const
433{
434 PRECICE_ASSERT(!hasMesh(mesh));
435 PRECICE_ASSERT(!_meshContexts.empty());
436
437 if (_meshContexts.size() == 1) {
438 return " This participant only knows mesh \"" + _meshContexts.begin()->first + "\".";
439 }
440
441 auto matches = utils::computeMatches(mesh, _meshContexts | boost::adaptors::map_keys);
442 if (matches.front().distance < 3) {
443 return " Did you mean mesh \"" + matches.front().name + "\"?";
444 } else {
445 return fmt::format(" Available meshes are: {}", fmt::join(_meshContexts | boost::adaptors::map_keys, ", "));
446 }
447}

449std::string ParticipantState::hintForMeshData(std::string_view mesh, std::string_view data) const
450{
451 PRECICE_ASSERT(hasMesh(mesh));
452 PRECICE_ASSERT(!hasData(mesh, data));
453 PRECICE_ASSERT(!_meshContexts.empty());
454
455 // Is there such data in other meshes?
456 std::vector<std::string> otherMeshes;
457 for (const auto &[_, mc] : _meshContexts) {
458 if (mc->mesh->hasDataName(data)) {
459 return " Did you mean the data of mesh \"" + mc->mesh->getName() + "\"?";
460 }
461 }
462
463 // Is there other data in the given mesh?
464 auto localData = meshContext(mesh).mesh->availableData();
465
466 if (localData.size() == 1) {
467 return " This mesh only knows data \"" + localData.front() + "\".";
468 }
469
470 // Was the data typoed?
471 auto matches = utils::computeMatches(data, localData);
472 if (matches.front().distance < 3) {
473 return " Did you mean data \"" + matches.front().name + "\"?";
474 }
475
476 return fmt::format(" Available data are: {}", fmt::join(localData, ", "));
477}
