373 void ParticipantConfiguration::finishParticipantConfiguration(
374  const xml::ConfigurationContext &context,
375  const impl::PtrParticipant & participant)
376 {
377  PRECICE_TRACE(participant->getName());
378 
379  // Set input/output meshes for data mappings and mesh requirements
380  using ConfMapping = mapping::MappingConfiguration::ConfiguredMapping;
381  for (const ConfMapping &confMapping : _mappingConfig->mappings()) {
382 
383  checkIllDefinedMappings(confMapping, participant);
384 
385  auto fromMesh = confMapping.fromMesh->getName();
386  auto toMesh = confMapping.toMesh->getName();
387 
388  if (confMapping.direction == mapping::MappingConfiguration::Direction::READ) {
390  PRECICE_CHECK(participant->isMeshReceived(fromMesh),
391  "Participant \"{}\" has a read mapping from mesh \"{}\", without receiving it. "
392  "Please add a receive-mesh tag with name=\"{}\"",
393  participant->getName(), fromMesh, fromMesh);
394  PRECICE_CHECK(participant->isMeshProvided(toMesh),
395  "Participant \"{}\" has a read mapping to mesh \"{}\", without providing it. "
396  "Please add a provide-mesh tag with name=\"{}\"",
397  participant->getName(), toMesh, toMesh);
398  } else {
399  // A write mapping maps from provided to received
400  PRECICE_CHECK(participant->isMeshProvided(fromMesh),
401  "Participant \"{}\" has a write mapping from mesh \"{}\", without providing it. "
402  "Please add a provided-mesh tag with name=\"{}\"",
403  participant->getName(), fromMesh, fromMesh);
404  PRECICE_CHECK(participant->isMeshReceived(toMesh),
405  "Participant \"{}\" has a write mapping to mesh \"{}\", without receiving it. "
406  "Please add a receive-mesh tag with name=\"{}\"",
407  participant->getName(), toMesh, toMesh);
408  }
409 
410  if (context.size > 1) {
411  if ((confMapping.direction == mapping::MappingConfiguration::WRITE &&
412  confMapping.mapping->getConstraint() == mapping::Mapping::CONSISTENT) ||
413  (confMapping.direction == mapping::MappingConfiguration::READ &&
414  confMapping.mapping->getConstraint() == mapping::Mapping::CONSERVATIVE)) {
415  PRECICE_ERROR("For a parallel participant, only the mapping combinations read-consistent and write-conservative are allowed");
416  } else if (confMapping.mapping->isScaledConsistent()) {
417  PRECICE_ERROR("Scaled consistent mapping is not yet supported for a parallel participant. "
418  "You could run in serial or use a plain (read-)consistent mapping instead.");
419  }
420  }
421 
422  const auto & fromMeshID = confMapping.fromMesh->getID();
423  const auto & toMeshID = confMapping.toMesh->getID();
424  impl::MeshContext &fromMeshContext = participant->meshContext(fromMesh);
425  impl::MeshContext &toMeshContext = participant->meshContext(toMesh);
426 
427  if (confMapping.direction == mapping::MappingConfiguration::READ) {
428  PRECICE_CHECK(toMeshContext.provideMesh,
429  "A read mapping of participant \"{}\" needs to map TO a provided mesh. Mesh \"{1}\" is not provided. "
430  "Please add the tag <provide-mesh name=\"{1}\" /> to the participant.",
431  participant->getName(), confMapping.toMesh->getName());
432  PRECICE_CHECK(not fromMeshContext.receiveMeshFrom.empty(),
433  "A read mapping of participant \"{}\" needs to map FROM a received mesh. Mesh \"{1}\" is not received. "
434  "Please add the tag <receive-mesh name=\"{1}\" /> to the participant.",
435  participant->getName(), confMapping.toMesh->getName());
436  } else {
437  PRECICE_CHECK(fromMeshContext.provideMesh,
438  "A write mapping of participant \"{}\" needs to map FROM a provided mesh. Mesh \"{1}\" is not provided. "
439  "Please add the tag <provide-mesh name=\"{1}\" /> to the participant.",
440  participant->getName(), confMapping.fromMesh->getName());
441  PRECICE_CHECK(not toMeshContext.receiveMeshFrom.empty(),
442  "A write mapping of participant \"{}\" needs to map TO a received mesh. Mesh \"{1}\" is not received. "
443  "Please add the tag <receive-mesh name=\"{1}\" /> to the participant.",
444  participant->getName(), confMapping.toMesh->getName());
445  }
446 
447  // @TODO: is this still correct?
448  if (confMapping.requiresBasisFunction) {
449  fromMeshContext.geoFilter = partition::ReceivedPartition::GeometricFilter::NO_FILTER;
450  toMeshContext.geoFilter = partition::ReceivedPartition::GeometricFilter::NO_FILTER;
451  }
452 
453  precice::impl::MappingContext mappingContext;
454  mappingContext.fromMeshID = fromMeshID;
455  mappingContext.toMeshID = toMeshID;
456 
457  mapping::PtrMapping &map = mappingContext.mapping;
458  PRECICE_ASSERT(map.get() == nullptr);
459  map = confMapping.mapping;
460  mappingContext.configuredWithAliasTag = confMapping.configuredWithAliasTag;
461 
462  const mesh::PtrMesh &input = fromMeshContext.mesh;
463  const mesh::PtrMesh &output = toMeshContext.mesh;
464  PRECICE_DEBUG("Configure mapping for input={}, output={}", input->getName(), output->getName());
465  map->setMeshes(input, output);
466 
467  if (confMapping.direction == mapping::MappingConfiguration::WRITE) {
468  participant->addWriteMappingContext(mappingContext);
469  } else {
470  PRECICE_ASSERT(confMapping.direction == mapping::MappingConfiguration::READ);
471  participant->addReadMappingContext(mappingContext);
472  }
473 
474  fromMeshContext.meshRequirement = std::max(
475  fromMeshContext.meshRequirement, map->getInputRequirement());
476  toMeshContext.meshRequirement = std::max(
477  toMeshContext.meshRequirement, map->getOutputRequirement());
478 
479  fromMeshContext.fromMappingContexts.push_back(mappingContext);
480  toMeshContext.toMappingContexts.push_back(mappingContext);
481  }
482  _mappingConfig->resetMappings();
483 
484  // Iterate over all write mappings
485  for (impl::MappingContext &mappingContext : participant->writeMappingContexts()) {
486  // Check, whether we can find a corresponding write data context
487  bool dataFound = false;
488  for (auto &dataContext : participant->writeDataContexts()) {
489  // First we look for the "from" mesh ID
490  const int fromMeshID = dataContext.getMeshID();
491  if (mappingContext.fromMeshID == fromMeshID) {
492  // Second we look for the "to" mesh ID
493  impl::MeshContext &meshContext = participant->meshContext(mappingContext.mapping->getOutputMesh()->getName());
494  // If this is true, we actually found a proper configuration
495  // If it is false, we look for another "from" mesh ID, because we might have multiple read and write mappings
496  if (meshContext.mesh->hasDataName(dataContext.getDataName())) {
497  // Check, if the fromMesh is a provided mesh
498  PRECICE_CHECK(participant->isMeshProvided(dataContext.getMeshName()),
499  "Participant \"{}\" has to provide mesh \"{}\" to be able to write data to it. "
500  "Please add a provide-mesh node with name=\"{}\".",
501  participant->getName(), dataContext.getMeshName(), dataContext.getMeshName());
502  dataContext.appendMappingConfiguration(mappingContext, meshContext);
503  // Enable gradient data if required
504  if (mappingContext.mapping->requiresGradientData() == true) {
505  mappingContext.requireGradientData(dataContext.getDataName());
506  }
507  dataFound = true;
508  }
509  }
510  }
511  PRECICE_CHECK(dataFound,
512  "Participant \"{}\" defines a write mapping from mesh \"{}\" to mesh \"{}\", "
513  "but there is either no corresponding write-data tag or the meshes used "
514  "by this participant lack the necessary use-data tags.",
515  participant->getName(), mappingContext.mapping->getInputMesh()->getName(), mappingContext.mapping->getOutputMesh()->getName());
516  }
517 
518  // Iterate over all read mappings
519  for (impl::MappingContext &mappingContext : participant->readMappingContexts()) {
520  // Check, weather we can find a corresponding read data context
521  bool dataFound = false;
522  for (auto &dataContext : participant->readDataContexts()) {
523  // First we look for the "to" mesh ID
524  const int toMeshID = dataContext.getMeshID();
525  if (mappingContext.toMeshID == toMeshID) {
526  // Second we look for the "from" mesh ID
527  impl::MeshContext &meshContext = participant->meshContext(mappingContext.mapping->getInputMesh()->getName());
528  // If this is true, we actually found a proper configuration
529  // If it is false, we look for another "from" mesh ID, because we might have multiple read and write mappings
530  if (meshContext.mesh->hasDataName(dataContext.getDataName())) {
531  // Check, if the toMesh is a provided mesh
532  PRECICE_CHECK(participant->isMeshProvided(dataContext.getMeshName()),
533  "Participant \"{}\" has to provide mesh \"{}\" in order to read data from it. "
534  "Please add a provide-mesh node with name=\"{}\".",
535  participant->getName(), dataContext.getMeshName(), dataContext.getMeshName());
536  dataContext.appendMappingConfiguration(mappingContext, meshContext);
537  // Enable gradient data if required
538  if (mappingContext.mapping->requiresGradientData() == true) {
539  mappingContext.requireGradientData(dataContext.getDataName());
540  }
541  dataFound = true;
542  }
543  }
544  }
545  PRECICE_CHECK(dataFound,
546  "Participant \"{}\" defines a read mapping from mesh \"{}\" to mesh \"{}\", "
547  "but there is either no corresponding read-data tag or the meshes used "
548  "by this participant lack the necessary use-data tags.",
549  participant->getName(), mappingContext.mapping->getInputMesh()->getName(), mappingContext.mapping->getOutputMesh()->getName());
550  }
551 
552  // Add actions
553  for (const action::PtrAction &action : _actionConfig->actions()) {
554  bool used = _participants.back()->isMeshUsed(action->getMesh()->getName());
555  PRECICE_CHECK(used,
556  "Data action of participant \"{}\" uses mesh \"{}\", which is not used by the participant. "
557  "Please add a provide-mesh or receive-mesh node with name=\"{}\".",
558  _participants.back()->getName(), action->getMesh()->getName(), action->getMesh()->getName());
559  }
560  for (action::PtrAction &action : _actionConfig->extractActions()) {
561  _participants.back()->addAction(std::move(action));
562  }
563 
564  // Add export contexts
565  for (io::ExportContext &exportContext : _exportConfig->exportContexts()) {
566  io::PtrExport exporter;
567  if (exportContext.type == VALUE_VTK) {
568  // This is handled with respect to the current configuration context.
569  // Hence, this is potentially wrong for every participant other than context.name.
570  if (context.size > 1) {
571  // Only display the warning message if this participant configuration is the current one.
572  if (context.name == participant->getName()) {
573  PRECICE_ERROR("You attempted to use the legacy VTK exporter with the parallel participant {}, which isn't supported."
574  "Migrate to another exporter, such as the VTU exporter by specifying \"<export:vtu ... />\" instead of \"<export:vtk ... />\".",
575  participant->getName());
576  }
577  } else {
578  exporter = io::PtrExport(new io::ExportVTK());
579  }
580  } else if (exportContext.type == VALUE_VTU) {
581  exporter = io::PtrExport(new io::ExportVTU());
582  } else if (exportContext.type == VALUE_VTP) {
583  exporter = io::PtrExport(new io::ExportVTP());
584  } else if (exportContext.type == VALUE_CSV) {
585  exporter = io::PtrExport(new io::ExportCSV());
586  } else {
587  PRECICE_ERROR("Participant {} defines an <export/> tag of unknown type \"{}\".",
588  _participants.back()->getName(), exportContext.type);
589  }
590  exportContext.exporter = std::move(exporter);
591 
592  _participants.back()->addExportContext(exportContext);
593  }
594  _exportConfig->resetExports();
595 
596  // Create watch points
597  for (const WatchPointConfig &config : _watchPointConfigs) {
598  PRECICE_CHECK(participant->isMeshUsed(config.nameMesh),
599  "Participant \"{}\" defines watchpoint \"{}\" for mesh \"{}\" which is not provided by the participant. "
600  "Please add <provide-mesh name=\"{}\" /> to the participant.",
601  participant->getName(), config.name, config.nameMesh, config.nameMesh);
602  const auto &meshContext = participant->usedMeshContext(config.nameMesh);
603  PRECICE_CHECK(meshContext.provideMesh,
604  "Participant \"{}\" defines watchpoint \"{}\" for the received mesh \"{}\", which is not allowed. "
605  "Please move the watchpoint definition to the participant providing mesh \"{}\".",
606  participant->getName(), config.name, config.nameMesh, config.nameMesh);
607 
608  std::string filename = "precice-" + participant->getName() + "-watchpoint-" + config.name + ".log";
609  participant->addWatchPoint(std::make_shared<impl::WatchPoint>(config.coordinates, meshContext.mesh, std::move(filename)));
610  }
611  _watchPointConfigs.clear();
612 
613  // Create watch integrals
614  for (const WatchIntegralConfig &config : _watchIntegralConfigs) {
615  PRECICE_CHECK(participant->isMeshUsed(config.nameMesh),
616  "Participant \"{}\" defines watch integral \"{}\" for mesh \"{}\" which is not used by the participant. "
617  "Please add a provide-mesh node with name=\"{}\".",
618  participant->getName(), config.name, config.nameMesh, config.nameMesh);
619  const auto &meshContext = participant->usedMeshContext(config.nameMesh);
620  PRECICE_CHECK(meshContext.provideMesh,
621  "Participant \"{}\" defines watch integral \"{}\" for the received mesh \"{}\", which is not allowed. "
622  "Please move the watchpoint definition to the participant providing mesh \"{}\".",
623  participant->getName(), config.name, config.nameMesh, config.nameMesh);
624 
625  std::string filename = "precice-" + participant->getName() + "-watchintegral-" + config.name + ".log";
626  participant->addWatchIntegral(std::make_shared<impl::WatchIntegral>(meshContext.mesh, std::move(filename), config.isScalingOn));
627  }
628  _watchIntegralConfigs.clear();
629 
630  // create default primary communication if needed
631  if (context.size > 1 && not _isIntraCommDefined && participant->getName() == context.name) {
632 #ifdef PRECICE_NO_MPI
633  PRECICE_ERROR("Implicit intra-participant communications for parallel participants are only available if preCICE was built with MPI. "
634  "Either explicitly define an intra-participant communication for each parallel participant or rebuild preCICE with \"PRECICE_MPICommunication=ON\".");
635 #else
636  utils::IntraComm::getCommunication() = std::make_shared<com::MPIDirectCommunication>();
637  participant->setUsePrimaryRank(true);
638 #endif
639  }
640  _isIntraCommDefined = false; // to not mess up with previous participant
641 }

643 void ParticipantConfiguration::checkIllDefinedMappings(
644  const mapping::MappingConfiguration::ConfiguredMapping &mapping,
645  const impl::PtrParticipant & participant)
646 {
647  PRECICE_TRACE();
648  using ConfMapping = mapping::MappingConfiguration::ConfiguredMapping;
649 
650  for (const ConfMapping &configuredMapping : _mappingConfig->mappings()) {
651  bool sameToMesh = mapping.toMesh->getName() == configuredMapping.toMesh->getName();
652  bool sameFromMesh = mapping.fromMesh->getName() == configuredMapping.fromMesh->getName();
653  if (sameToMesh && sameFromMesh) {
654  // It's really the same mapping, not a duplicated one. Those are already checked for in MappingConfiguration.
655  return;
656  }
657 
658  if (sameToMesh) {
659  for (const mesh::PtrData &data : mapping.fromMesh->data()) {
660  for (const mesh::PtrData &configuredData : configuredMapping.fromMesh->data()) {
661  bool sameFromData = data->getName() == configuredData->getName();
662 
663  if (not sameFromData) {
664  continue;
665  }
666 
667  bool sameDirection = false;
668 
669  if (mapping.direction == mapping::MappingConfiguration::WRITE) {
670  for (const auto &dataContext : participant->writeDataContexts()) {
671  sameDirection |= data->getName() == dataContext.getDataName();
672  }
673  }
674  if (mapping.direction == mapping::MappingConfiguration::READ) {
675  for (const auto &dataContext : participant->readDataContexts()) {
676  sameDirection |= data->getName() == dataContext.getDataName();
677  }
678  }
679  PRECICE_CHECK(!sameDirection,
680  "There cannot be two mappings to mesh \"{}\" if the meshes from which is mapped contain "
681  "duplicated data fields that are also actually mapped on this participant. "
682  "Here, both from meshes contain data \"{}\". "
683  "The mapping is not well defined. "
684  "Which data \"{}\" should be mapped to mesh \"{}\"?",
685  mapping.toMesh->getName(), data->getName(), data->getName(), mapping.toMesh->getName());
686  }
687  }
688  }
689  }
690 }
