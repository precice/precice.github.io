87inline double computeSquaredDifference(
88 const std::array<double, 3> &u,
89 std::array<double, 3> v,
90 const std::array<bool, 3> & activeAxis = {{true, true, true}})
91{
92 // Subtract the values and multiply out dead dimensions
93 for (unsigned int d = 0; d < v.size(); ++d) {
94 v[d] = (u[d] - v[d]) * static_cast<int>(activeAxis[d]);
95 }
96 // @todo: this can be replaced by std::hypot when moving to C++17
97 return std::accumulate(v.begin(), v.end(), static_cast<double>(0.), [](auto &res, auto &val) { return res + val * val; });
98}

101template <typename IndexContainer>
102constexpr void reduceActiveAxis(const mesh::Mesh &mesh, const IndexContainer &IDs, std::array<bool, 3> &axis)
103{
104 // make a pair of the axis and the difference
105 std::array<std::pair<int, double>, 3> differences;
106
107 // Compute the difference magnitude per direction
108 for (std::size_t d = 0; d < axis.size(); ++d) {
109 // Ignore dead axis here, i.e., apply the max value such that they are sorted on the last position(s)
110 if (axis[d] == false) {
111 differences[d] = std::make_pair<int, double>(d, std::numeric_limits<double>::max());
112 } else {
113 auto res = std::minmax_element(IDs.begin(), IDs.end(), [&](const auto &a, const auto &b) { return mesh.vertex(a).coord(d) < mesh.vertex(b).coord(d); });
114 // Check if we are above or below the threshold
115 differences[d] = std::make_pair<int, double>(d, std::abs(mesh.vertex(*res.second).coord(d) - mesh.vertex(*res.first).coord(d)));
116 }
117 }
118
119 std::sort(differences.begin(), differences.end(), [](const auto &d1, const auto &d2) { return d1.second < d2.second; });
120 // Disable the axis having the smallest expansion
121 axis[differences[0].first] = false;
122}

125template <typename IndexContainer>
126inline void fillPolynomialEntries(Eigen::MatrixXd &matrix, const mesh::Mesh &mesh, const IndexContainer &IDs, Eigen::Index startIndex, std::array<bool, 3> activeAxis)
127{
128 // Loop over all vertices in the mesh
129 for (const auto &i : IDs | boost::adaptors::indexed()) {
130
131 // 1. the constant contribution
132 matrix(i.index(), startIndex) = 1.0;
133
134 // 2. the linear contribution
135 const auto & u = mesh.vertex(i.value()).rawCoords();
136 unsigned int k = 0;
137 // Loop over all three space dimension and ignore dead axis
138 for (unsigned int d = 0; d < activeAxis.size(); ++d) {
139 if (activeAxis[d]) {
140 PRECICE_ASSERT(matrix.rows() > i.index(), matrix.rows(), i.index());
141 PRECICE_ASSERT(matrix.cols() > startIndex + 1 + k, matrix.cols(), startIndex + 1 + k);
142 matrix(i.index(), startIndex + 1 + k) = u[d];
143 ++k;
144 }
145 }
146 }
147}

149template <typename RADIAL_BASIS_FUNCTION_T, typename IndexContainer>
150Eigen::MatrixXd buildMatrixCLU(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, const IndexContainer &inputIDs,
151 std::array<bool, 3> activeAxis, Polynomial polynomial)
152{
153 // Treat the 2D case as 3D case with dead axis
154 const unsigned int deadDimensions = std::count(activeAxis.begin(), activeAxis.end(), false);
155 const unsigned int dimensions = 3;
156 const unsigned int polyparams = polynomial == Polynomial::ON ? 1 + dimensions - deadDimensions : 0;
157
158 // Add linear polynom degrees if polynomial requires this
159 const auto inputSize = inputIDs.size();
160 const auto n = inputSize + polyparams;
161
162 PRECICE_ASSERT((inputMesh.getDimensions() == 3) || activeAxis[2] == false);
163 PRECICE_ASSERT((inputSize >= 1 + polyparams) || polynomial != Polynomial::ON, inputSize);
164
165 Eigen::MatrixXd matrixCLU(n, n);
166
167 // Required to fill the poly -> poly entries in the matrix, which remain otherwise untouched
168 if (polynomial == Polynomial::ON) {
169 matrixCLU.setZero();
170 }
171
172 // Compute RBF matrix entries
173 auto i_iter = inputIDs.begin();
174 Eigen::Index i_index = 0;
175 for (; i_iter != inputIDs.end(); ++i_iter, ++i_index) {
176 const auto &u = inputMesh.vertex(*i_iter).rawCoords();
177 auto j_iter = i_iter;
178 auto j_index = i_index;
179 for (; j_iter != inputIDs.end(); ++j_iter, ++j_index) {
180 const auto &v = inputMesh.vertex(*j_iter).rawCoords();
181 double squaredDifference = computeSquaredDifference(u, v, activeAxis);
182 matrixCLU(i_index, j_index) = basisFunction.evaluate(std::sqrt(squaredDifference));
183 }
184 }
185
186 // Add potentially the polynomial contribution in the matrix
187 if (polynomial == Polynomial::ON) {
188 fillPolynomialEntries(matrixCLU, inputMesh, inputIDs, inputSize, activeAxis);
189 }
190 matrixCLU.triangularView<Eigen::Lower>() = matrixCLU.transpose();
191 return matrixCLU;
192}

229// Variant operating on the Cholesky decopmosition
230inline Eigen::VectorXd computeInverseDiagonal(Eigen::LLT<Eigen::MatrixXd> decMatrixC)
231{
232 // 1. Compute the diagonal entries of the inverse kernel matrix:
233 // We already have the Cholesky decomposition. So instead of solving for the
234 // kernel matrix directly, we invert the lower triangular matrix of the
235 // decomposition:
236 // using A^{-1} = (L^T)^{-1}L^{-1} enables the computation of the diagonal
237 // entries of A^{-1} by evaluating the product above:
238 // A^{-1}_{ii} = sum_{k=1}^n L^{-T}_{ik} L^{-1}_{ki}
239
240 // 1a: Compute the inverse of the lower triangular matrix L
241 // Eigen::MatrixXd L_inv = L.inverse(); is not supported by Eigen (linker errors)
242 // However, Eigen provides triangular solver (LAPACK::trsm), which can be used
243 // to solve L * Linv = I
244 // Eigen::MatrixXd L_inv = Eigen::MatrixXd::Identity(inSize, inSize);
245 // _decMatrixC.matrixL().solveInPlace(L_inv);
246 // which yields cubic complexity (BLAS level 3).
247
248 // Here, we use our own implementation to compute the triangular inverse
249 // (similar to LAPACK:trtri) more efficient, given that the RHS is also
250 // triangular was unfortunately slower.
251
252 // Solve L * Linv = I
253 // Eigen::MatrixXd L_inv = Eigen::MatrixXd::Identity(n, n);
254 // decMatrixC.matrixL().solveInPlace(L_inv);
255 Eigen::MatrixXd L_inv = utils::invertLowerTriangularBlockwise(decMatrixC.matrixL());
256
257 // 1b: Compute the diagonal elements of A^{-1} by evaluating (L^T)^{-1}L^{-1}
258 Eigen::VectorXd inverseDiagonal = (L_inv.array().square().colwise().sum()).transpose();
259
260 return inverseDiagonal;
261}

263// Variant operating on the QR decomposition
264inline Eigen::VectorXd computeInverseDiagonal(Eigen::ColPivHouseholderQR<Eigen::MatrixXd> decMatrixC)
265{
266 // 1. Compute the diagonal entries of the inverse kernel matrix:
267 // We could use
268 // diag_inv_A= _decMatrixC.inverse().diagonal();
269 // as Eigen offers this for the QR decomposition, but the inverse() call is
270 // more expensive than it has to be (as we compute all entries of the inverse). On the
271 // other hand, using non strictily-positive definite functions is less relevant anyway.
272 // Still, let's try the following:
273
274 // We already have the QR decomposition. So instead of solving for the
275 // kernel matrix directly, make use of the following:
276 // A^{-1} = R^{-1}Q^{-1}
277 // Since Q is orthogonal, Q^{-1} = Q^T
278 // R is upper triangular and we need to compute the inverse (using backwards substitution)
279
280 // enables the computation of the diagonal
281 // entries of A^{-1} by evaluating the product above:
282 // A^{-1} = R^{-1} Q^T
283 // A^{-1}_{ii} = sum_{k=1}^n R^{-1}_{ik} Q^{T}_{ki}
284
285 Eigen::VectorXd inverseDiagonal;
286 const Eigen::Index n = decMatrixC.matrixR().cols();
287
288 // 1a: Compute the inverse of the lower triangular matrix L
289 // Solve R * Rinv = I
290 Eigen::MatrixXd R_inv = Eigen::MatrixXd::Identity(n, n);
291 decMatrixC.matrixR().template triangularView<Eigen::Upper>().solveInPlace(R_inv);
292 Eigen::VectorXi P = decMatrixC.colsPermutation().indices();
293 Eigen::MatrixXd Q = decMatrixC.householderQ();
294
295 // Now evaluate, caution with the column permutation
296 inverseDiagonal.resize(n);
297 for (Eigen::Index i = 0; i < n; ++i) {
298 inverseDiagonal(P(i)) = R_inv.row(i) * Q.transpose().col(P(i));
299 }
300 return inverseDiagonal;
301}

303template <typename RADIAL_BASIS_FUNCTION_T>
304double RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::evaluateRippaLOOCVerror(const Eigen::VectorXd &lambda) const
305{
306 // Implementation of LOOCV according to Rippa(1999), DOI: 10.1023/a:1018975909870
307 double loocv = 0;
308 const Eigen::Index n = lambda.size();
309 // 2: Next, compute the RBF coefficient. These are exactly the same as computed during
310 // the solve_consistent and we could also pass them into the function, depending on how
311 // often wen want to compute the LOOCV. We omit the polynomial contribution here, i.e.,
312 // the input data remains untouched
313 // Eigen::VectorXd lambda = _decMatrixC.solve(inputData);
314
315 // 3: Evaluate the Rippa formula:
316 // The error estimate is given by a component-wise division: lambda/A^{-1}_{ii}.
317 // We then compute the RMS of all LOOCV error entries (other options for the
318 // aggregation should be possible)
319 // TODO:Consider storing the reciprocal inverse diagonal entries
320 loocv = std::sqrt((lambda.array() / _inverseDiagonal.array()).array().square().sum() / n);
321
322 return loocv;
323}

325template <typename RADIAL_BASIS_FUNCTION_T>
326template <typename IndexContainer>
327RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::RadialBasisFctSolver(RADIAL_BASIS_FUNCTION_T basisFunction, const mesh::Mesh &inputMesh, const IndexContainer &inputIDs,
328 const mesh::Mesh &outputMesh, const IndexContainer &outputIDs, std::vector<bool> deadAxis, Polynomial polynomial)
329{
330 PRECICE_ASSERT(!(RADIAL_BASIS_FUNCTION_T::isStrictlyPositiveDefinite() && polynomial == Polynomial::ON), "The integrated polynomial (polynomial=\"on\") is not supported for the selected radial-basis function. Please select another radial-basis function or change the polynomial configuration.");
331 // Convert dead axis vector into an active axis array so that we can handle the reduction more easily
332 std::array<bool, 3> activeAxis({{false, false, false}});
333 std::transform(deadAxis.begin(), deadAxis.end(), activeAxis.begin(), [](const auto ax) { return !ax; });
334
335 // First, assemble the interpolation matrix and check the invertability
336 bool decompositionSuccessful = false;
337 if constexpr (RADIAL_BASIS_FUNCTION_T::isStrictlyPositiveDefinite()) {
338 _decMatrixC = buildMatrixCLU(basisFunction, inputMesh, inputIDs, activeAxis, polynomial).llt();
339 decompositionSuccessful = _decMatrixC.info() == Eigen::ComputationInfo::Success;
340 } else {
341 _decMatrixC = buildMatrixCLU(basisFunction, inputMesh, inputIDs, activeAxis, polynomial).colPivHouseholderQr();
342 decompositionSuccessful = _decMatrixC.isInvertible();
343 }
344
345 PRECICE_CHECK(decompositionSuccessful,
346 "The interpolation matrix of the RBF mapping from mesh \"{}\" to mesh \"{}\" is not invertable. "
347 "This means that the mapping problem is not well-posed. "
348 "Please check if your coupling meshes are correct (e.g. no vertices are duplicated) or reconfigure "
349 "your basis-function (e.g. reduce the support-radius).",
350 inputMesh.getName(), outputMesh.getName());
351
352 // For polynomial on, the algorithm might fail in determining the size of the system
353 if (polynomial != Polynomial::ON && computeCrossValidation) {
354 // TODO: Disable synchronization
355 precice::profiling::Event e("map.rbf.computeLOOCV");
356 _inverseDiagonal = computeInverseDiagonal(_decMatrixC);
357 }
358 // Second, assemble evaluation matrix
359 _matrixA = buildMatrixA(basisFunction, inputMesh, inputIDs, outputMesh, outputIDs, activeAxis, polynomial);
360
361 // In case we deal with separated polynomials, we need dedicated matrices for the polynomial contribution
362 if (polynomial == Polynomial::SEPARATE) {
363
364 // 4 = 1 + dimensions(3) = maximum number of polynomial parameters
365 auto localActiveAxis = activeAxis;
366 unsigned int polyParams = 4 - std::count(localActiveAxis.begin(), localActiveAxis.end(), false);
367
368 do {
369 // First, build matrix Q and check for the condition number
370 _matrixQ.resize(inputIDs.size(), polyParams);
371 fillPolynomialEntries(_matrixQ, inputMesh, inputIDs, 0, localActiveAxis);
372
373 // Compute the condition number
374 Eigen::JacobiSVD<Eigen::MatrixXd> svd(_matrixQ);
375 PRECICE_ASSERT(svd.singularValues().size() > 0);
376 PRECICE_DEBUG("Singular values in polynomial solver: {}", svd.singularValues());
377 const double conditionNumber = svd.singularValues()(0) / std::max(svd.singularValues()(svd.singularValues().size() - 1), math::NUMERICAL_ZERO_DIFFERENCE);
378 PRECICE_DEBUG("Condition number: {}", conditionNumber);
379
380 // Disable one axis
381 if (conditionNumber > 1e5) {
382 reduceActiveAxis(inputMesh, inputIDs, localActiveAxis);
383 polyParams = 4 - std::count(localActiveAxis.begin(), localActiveAxis.end(), false);
384 } else {
385 break;
386 }
387 } while (true);
388
389 // allocate and fill matrix V for the outputMesh
390 _matrixV.resize(outputIDs.size(), polyParams);
391 fillPolynomialEntries(_matrixV, outputMesh, outputIDs, 0, localActiveAxis);
392
393 // 3. compute decomposition
394 _qrMatrixQ = _matrixQ.colPivHouseholderQr();
395 }
396}

398template <typename RADIAL_BASIS_FUNCTION_T>
399Eigen::VectorXd RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::solveConservative(const Eigen::VectorXd &inputData, Polynomial polynomial) const
400{
401 PRECICE_ASSERT((_matrixV.size() > 0 && polynomial == Polynomial::SEPARATE) || _matrixV.size() == 0, _matrixV.size());
402 // TODO: Avoid temporary allocations
403 // Au is equal to the eta in our PETSc implementation
404 PRECICE_ASSERT(inputData.size() == _matrixA.rows());
405 Eigen::VectorXd Au = _matrixA.transpose() * inputData;
406 PRECICE_ASSERT(Au.size() == _matrixA.cols());
407
408 // mu in the PETSc implementation
409 Eigen::VectorXd out = _decMatrixC.solve(Au);
410
411 if (polynomial == Polynomial::SEPARATE) {
412 Eigen::VectorXd epsilon = _matrixV.transpose() * inputData;
413 PRECICE_ASSERT(epsilon.size() == _matrixV.cols());
414
415 // epsilon = Q^T * mu - epsilon (tau in the PETSc impl)
416 epsilon -= _matrixQ.transpose() * out;
417 PRECICE_ASSERT(epsilon.size() == _matrixQ.cols());
418
419 // out = out - solveTranspose tau (sigma in the PETSc impl)
420 out -= static_cast<Eigen::VectorXd>(_qrMatrixQ.transpose().solve(-epsilon));
421 }
422 return out;
423}

425template <typename RADIAL_BASIS_FUNCTION_T>
426Eigen::VectorXd RadialBasisFctSolver<RADIAL_BASIS_FUNCTION_T>::solveConsistent(Eigen::VectorXd &inputData, Polynomial polynomial) const
427{
428 PRECICE_ASSERT((_matrixQ.size() > 0 && polynomial == Polynomial::SEPARATE) || _matrixQ.size() == 0);
429 Eigen::VectorXd polynomialContribution;
430 // Solve polynomial QR and subtract it from the input data
431 if (polynomial == Polynomial::SEPARATE) {
432 polynomialContribution = _qrMatrixQ.solve(inputData);
433 inputData -= (_matrixQ * polynomialContribution);
434 }
435
436 // Integrated polynomial (and separated)
437 PRECICE_ASSERT(inputData.size() == _matrixA.cols());
438 Eigen::VectorXd p = _decMatrixC.solve(inputData);
439
440 if (polynomial != Polynomial::ON && computeCrossValidation) {
441 precice::profiling::Event e("map.rbf.evaluateLOOCV");
442 PRECICE_INFO("Cross validation error (LOOCV): {}", evaluateRippaLOOCVerror(p));
443 }
444 PRECICE_ASSERT(p.size() == _matrixA.cols());
445 Eigen::VectorXd out = _matrixA * p;
446
447 // Add the polynomial part again for separated polynomial
448 if (polynomial == Polynomial::SEPARATE) {
449 out += (_matrixV * polynomialContribution);
450 }
451 return out;
452}
